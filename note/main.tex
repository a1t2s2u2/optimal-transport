\begin{abstract}
最適輸送理論は，ある確率分布を別の確率分布へ移すための最小コストを求める問題を扱う．
本稿では，Monge~\cite{Monge1781} による原初的な定式化から出発し，
Kantorovich~\cite{Kantorovich1942} による緩和とその双対理論を展開した後，
離散問題に対する線形計画法（単体法）による系統的な解法を示す．
\end{abstract}

\section{Monge 問題}

確率分布の再配置を最適化する問題は，
フランスの数学者 Monge が土木工学の文脈で提起したことに始まる．
与えられた二つの確率分布に対し，
一方を他方へ移すための最小費用を達成する写像を求めることが目標である．
この問題を正確に記述するには，
「確率分布」「写像による分布の移動」を測度論の言葉で定式化する必要がある．
以下では位相空間・$\sigma$加法族・測度・押し出し測度の順に準備を整え，
Monge 問題の定義と基本的な計算例を示す（例~\ref{ex:monge_example}）．
続いて，特定の分布の組では Monge 写像が一切存在しない（例~\ref{ex:monge_infeasible}）
ことを具体的に確認し，第2節の Kantorovich 緩和の動機を与える．

\begin{definition}{位相空間}{topological_space}
集合 $X$ の部分集合族 $\mathcal{O}\subset 2^X$ が
$X$ 上の\textbf{位相}（topology）であるとは，
次の三条件を満たすことをいう：
\begin{enumerate}
  \item $\emptyset\in\mathcal{O}$ かつ $X\in\mathcal{O}$，
  \item $O_1,O_2\in\mathcal{O}$ ならば $O_1\cap O_2\in\mathcal{O}$（有限交叉で閉じる），
  \item $\{O_\lambda\}_{\lambda\in\Lambda}\subset\mathcal{O}$ ならば
        $\bigcup_{\lambda\in\Lambda}O_\lambda\in\mathcal{O}$（任意合併で閉じる）．
\end{enumerate}
組 $(X,\mathcal{O})$ を\textbf{位相空間}と呼び，
$\mathcal{O}$ の元を\textbf{開集合}という．
\end{definition}

\begin{definition}{$\sigma$加法族}{sigma_algebra}
集合 $X$ の部分集合族 $\mathcal{A}\subset 2^X$ が
\textbf{$\sigma$加法族}（$\sigma$-algebra）であるとは，
次の三条件を満たすことをいう：
\begin{enumerate}
  \item $X\in\mathcal{A}$，
  \item $A\in\mathcal{A}$ ならば $A^c\in\mathcal{A}$，
  \item $A_1,A_2,\ldots\in\mathcal{A}$ ならば
        $\bigcup_{n=1}^{\infty}A_n\in\mathcal{A}$．
\end{enumerate}
\end{definition}

\begin{definition}{可測空間}{measurable_space}
集合 $X$ とその上の $\sigma$加法族 $\mathcal{A}$ の組
$(X,\mathcal{A})$ を\textbf{可測空間}と呼ぶ．
$\mathcal{A}$ の元を\textbf{可測集合}という．
特に，位相空間 $(X,\mathcal{O})$ に対し，
開集合全体 $\mathcal{O}$ を部分集合として含む $\sigma$加法族のうち，
最小のものを\textbf{Borel $\sigma$加法族}と呼び $\mathcal{B}(X)$ と記す．
具体的には，$\mathcal{O}\subset\mathcal{F}$ を満たす
$\sigma$加法族 $\mathcal{F}$ をすべて集めたとき，
$\mathcal{B}(X)=\bigcap\mathcal{F}$ として得られる．
$\mathcal{B}(X)$ の元を\textbf{Borel 集合}という．
\end{definition}

\begin{example}{$\mathbb{R}$ 上のBorel $\sigma$加法族}{borel_real}
$X=\mathbb{R}$ の開集合は開区間やその合併（例：$(0,1)$，$(-\infty,3)\cup(5,7)$）である．
$\mathcal{B}(\mathbb{R})$ はこれらの開集合を含む最小の $\sigma$加法族であり，
$\sigma$加法族の性質から補集合や可算合併も含むため，
閉集合（例：$[0,1]$），半開区間（例：$[0,1)$），一点集合 $\{x\}$，
可算集合 $\mathbb{Q}$ なども Borel 集合である．
\end{example}

Borel 集合でない部分集合の典型例として\textbf{Vitali 集合}がある．
これは $[0,1]$ 上で同値関係 $x\sim y \Leftrightarrow x-y\in\mathbb{Q}$
を定め，各同値類から選択公理を用いて代表元を一つずつ選んで作る集合である．
この集合に対しては，測度を矛盾なく割り当てることができない
（平行移動不変性と $\sigma$加法性が両立しない）ため，
Borel 集合に属さない．

\begin{definition}{測度}{measure}
可測空間 $(X,\mathcal{A})$ 上の\textbf{測度}とは，
関数 $\mu\colon\mathcal{A}\to[0,\infty]$ であって次を満たすものをいう：
\begin{enumerate}
  \item $\mu(\emptyset)=0$，
  \item 互いに素な可測集合の列 $A_1,A_2,\ldots\in\mathcal{A}$
        （$A_i\cap A_j=\emptyset$, $i\neq j$）に対し
        \[
          \mu\!\left(\bigcup_{n=1}^{\infty}A_n\right)
          =\sum_{n=1}^{\infty}\mu(A_n)
        \]
        が成り立つ（$\sigma$加法性）．
\end{enumerate}
組 $(X,\mathcal{A},\mu)$ を\textbf{測度空間}という．
特に $\mu(X)=1$ を満たす測度を\textbf{確率測度}と呼ぶ．
$\sigma$加法族として Borel $\sigma$加法族 $\mathcal{B}(X)$ を選んだ
可測空間 $(X,\mathcal{B}(X))$ 上の確率測度を\textbf{Borel確率測度}という．
\end{definition}

\begin{definition}{可測写像}{measurable_map}
$(X,\mathcal{F})$，$(Y,\mathcal{G})$ を可測空間とする．
写像 $T\colon X\to Y$ が\textbf{可測写像}であるとは，
任意の $B\in\mathcal{G}$ に対し $T^{-1}(B)\in\mathcal{F}$ が成り立つことをいう．
\end{definition}

測度論では，集合の「大きさ」は $\sigma$加法族に属する集合に対してのみ定まる．
写像 $T$ を通じて $Y$ 側の集合 $B$ を $X$ 側で扱うには，
逆像 $T^{-1}(B)$ が $X$ の $\sigma$加法族 $\mathcal{A}$ に属する必要がある．

\begin{definition}{Borel写像}{borel_map}
$X$，$Y$ を位相空間とする．
写像 $T\colon X\to Y$ が\textbf{Borel写像}（Borel可測写像）であるとは，
$T$ が可測空間 $(X,\mathcal{B}(X))$ から $(Y,\mathcal{B}(Y))$ への可測写像であること，
すなわち任意の $B\in\mathcal{B}(Y)$ に対し $T^{-1}(B)\in\mathcal{B}(X)$ が成り立つことをいう．
\end{definition}

\begin{definition}{押し出し測度}{pushforward}
$(X,\mathcal{F})$，$(Y,\mathcal{G})$ を可測空間，
$\mu$ を $(X,\mathcal{F})$ 上の測度，$T\colon X\to Y$ を可測写像とする．
$T$ による $\mu$ の\textbf{押し出し測度} $T_\#\mu$ を
\[
  T_\#\mu(B) \coloneqq \mu(T^{-1}(B)), \qquad B\in\mathcal{G}
\]
で定義する．
\end{definition}

押し出し測度を用いると，Monge の最適輸送問題は次のように定式化される．

\begin{definition}{Monge 問題}{monge}
$X$，$Y$ を位相空間とし，
$\mu$，$\nu$ をそれぞれ $X$，$Y$ 上のBorel確率測度，
$c\colon X\times Y\to[0,\infty]$ をBorel写像とする．
\textbf{Monge 問題}は，$T_\#\mu=\nu$ を満たすBorel写像
$T\colon X\to Y$（\textbf{輸送写像}）の中で
\[
  \inf_{T:\,T_\#\mu=\nu}\int_{X} c\bigl(x,\,T(x)\bigr)\,d\mu(x)
\]
を達成するものを求める問題である．
\end{definition}

制約 $T_\#\mu=\nu$ は，$Y$ の任意のBorel集合 $B$ に対して
$\mu(T^{-1}(B))=\nu(B)$ が成り立つことを意味する．
すなわち，$B$ に送り込まれる分の $\mu$ での測度と，
$\nu$ が $B$ に割り当てる測度がちょうど一致する，
輸送の結果として過不足が生じないことを要求する条件である．

\begin{definition}{Dirac測度}{dirac}
$X$ を位相空間，$x_0\in X$ とする．
任意のBorel集合 $B\in\mathcal{B}(X)$ に対し
\[
  \delta_{x_0}(B) \coloneqq
  \begin{cases}
    1 & (x_0\in B), \\
    0 & (x_0\notin B)
  \end{cases}
\]
で定まるBorel確率測度 $\delta_{x_0}$ を，
$x_0$ に集中する\textbf{Dirac測度}という．
\end{definition}

\begin{remark}{Dirac測度はBorel確率測度である}{dirac_is_borel}
Dirac測度 $\delta_{x_0}$ はBorel確率測度であることを確認する．
\begin{itemize}
  \item $\delta_{x_0}(\emptyset)=0$：$x_0\notin\emptyset$ より成立．
  \item $\sigma$加法性：互いに素なBorel集合の列 $B_1,B_2,\ldots\in\mathcal{B}(X)$ に対し，
        $x_0$ はたかだか一つの $B_n$ にしか属さないので，
        $\delta_{x_0}(\bigcup_n B_n)$ と $\sum_n\delta_{x_0}(B_n)$ は
        ともに $x_0$ がいずれかの $B_n$ に属すれば $1$，そうでなければ $0$ で一致する．
  \item $\delta_{x_0}(X)=1$：$x_0\in X$ より成立（確率測度の条件）．
\end{itemize}
したがって $\delta_{x_0}$ はBorel確率測度である．
\end{remark}

\begin{proposition}{Dirac測度に関する積分}{integral_dirac}
$f\colon X\to\mathbb{R}$ をBorel写像とする．
Dirac測度 $\delta_{x_0}$ に関する積分は
\[
  \int_X f(x)\,d\delta_{x_0}(x) = f(x_0)
\]
で与えられる．
\end{proposition}

\begin{example}{Monge 問題の具体例}{monge_example}
$X=Y=\mathbb{R}$ とし，Dirac測度の重み付き和で確率測度を定める．
$\mu$ は地点 $0,\,1,\,2$ に，$\nu$ は地点 $4,\,6$ に集中するとする：
\[
  \mu = 0.5\,\delta_0 + 0.3\,\delta_1 + 0.2\,\delta_2, \qquad
  \nu = 0.8\,\delta_4 + 0.2\,\delta_6.
\]
これは，任意のBorel集合 $B$ に対し
\[
  \mu(B) = 0.5\cdot\delta_0(B) + 0.3\cdot\delta_1(B) + 0.2\cdot\delta_2(B)
\]
を意味する．各 $\delta$ が「その点が $B$ に含まれるか」を返すスイッチ，
係数が配分量として機能する．例えば
\[
  \mu(\{0,1\})
  = 0.5\cdot\underbrace{\delta_0(\{0,1\})}_{=1}
  + 0.3\cdot\underbrace{\delta_1(\{0,1\})}_{=1}
  + 0.2\cdot\underbrace{\delta_2(\{0,1\})}_{=0}
  = 0.8
\]
である．コスト関数を距離 $c(x,y)=|x-y|$（遠くに運ぶほどコストが高い）とする．

写像 $T_1$ を $T_1(0)=4$，$T_1(1)=4$，$T_1(2)=6$ と定めると，
\begin{align*}
  T_{1\#}\mu(\{4\}) &= \mu(\{0,1\}) = 0.8 = \nu(\{4\}), \\
  T_{1\#}\mu(\{6\}) &= \mu(\{2\}) = 0.2 = \nu(\{6\})
\end{align*}
より $T_{1\#}\mu=\nu$ が成り立ち，$T_1$ は輸送写像の制約を満たす．
このときの輸送コストを計算する．
$\mu = 0.5\,\delta_0 + 0.3\,\delta_1 + 0.2\,\delta_2$ であるから，
命題~\ref{prop:integral_dirac}（Dirac測度に関する積分）を各項に適用すると，
\begin{align*}
  \int |x-T_1(x)|\,d\mu
  &= 0.5\int|x-T_1(x)|\,d\delta_0
   + 0.3\int|x-T_1(x)|\,d\delta_1
   + 0.2\int|x-T_1(x)|\,d\delta_2 \\
  &= 0.5\cdot|0-T_1(0)| + 0.3\cdot|1-T_1(1)| + 0.2\cdot|2-T_1(2)| \\
  &= 0.5\cdot|0-4| + 0.3\cdot|1-4| + 0.2\cdot|2-6| \\
  &= 2.0 + 0.9 + 0.8 = 3.7
\end{align*}
である．
この例では Monge 写像 $T_1$ が存在したが，次の例が示すように，
分布の組によっては条件を満たす写像が一切存在しない場合がある．
\end{example}

直感的には，$\mu$ が送り元の分布，$\nu$ が送り先の分布を表し，
コスト $c(x,y)$ は地点 $x$ から地点 $y$ に運ぶ費用を意味する．
Monge 問題は，$\mu$ を $\nu$ にちょうど移すための総輸送コストを最小化する
最適な写像を求める問題である．

しかし Monge 問題には，
制約 $T_\#\mu=\nu$ を満たす輸送写像がそもそも存在しない場合がある
という根本的な困難がある．
以降の各節で繰り返し用いる具体例を導入する．

\begin{setting}{基本設定}{basic}
\[
  X=Y=\{0,1\},\quad
  \mu = \tfrac{3}{4}\delta_0 + \tfrac{1}{4}\delta_1,\quad
  \nu  = \tfrac{2}{4}\delta_0 + \tfrac{2}{4}\delta_1,\quad
  c(x,y) = |x-y|.
\]
\end{setting}

\begin{example}{Monge 問題の非可解性}{monge_infeasible}
設定~\ref{set:basic}に対し，
$T_\#\mu=\nu$ を満たす写像 $T:\{0,1\}\to\{0,1\}$ が存在するか，
$X=\{0,1\}$ は2点集合なので取りうる写像の全4通りを確認する：
\[
  \begin{array}{ll}
    T\equiv 0: & T_\#\mu = \delta_0 \ne \nu, \\[2pt]
    T\equiv 1: & T_\#\mu = \delta_1 \ne \nu, \\[2pt]
    T=\mathrm{id}: & T_\#\mu = \mu = \tfrac{3}{4}\delta_0+\tfrac{1}{4}\delta_1 \ne \nu, \\[2pt]
    T(0)=1,\,T(1)=0: & T_\#\mu = \tfrac{1}{4}\delta_0+\tfrac{3}{4}\delta_1 \ne \nu.
  \end{array}
\]
いずれも $T_\#\mu\ne\nu$ であり，Monge 問題は\textbf{実行不可能}である．
写像 $T$ は確率質量を分割できないため，
$\mu$ が $0$ に偏って分布している状況で $\nu$ のような均等分布を実現できない．
\end{example}

この例が示すように，Monge 問題は「写像」による輸送を要求するため，
測度の分割が必要な場面で実行不可能になる．
この本質的な制約を克服するため，
Kantorovich は輸送写像を輸送計画（結合測度）に緩和するという着想を導入した．

\begin{remark}{連続分布における Monge 問題}{monge_continuous}
離散分布では Monge 写像が存在しない場合があるが，
$\mu$ が Lebesgue 測度に関して絶対連続（密度関数を持つ）な場合は状況が異なる．
連続分布では $T$ が異なる $x$ を異なる $T(x)$ に割り当てられるため，
目標分布 $\nu$ を実現するような写像が構成しやすくなる．
この事実を厳密に保証するのが次の Brenier の定理であり，
第3節の $c$変換の構造（注意~\ref{rem:legendre_fenchel}）と深く関連する．

まとめると，Monge 問題の「実行不可能性」は主に離散設定に起因する問題であり，
Kantorovich 緩和はその障害を取り除く汎用的な枠組みである．
\end{remark}

\begin{theorem}{Brenier の定理~\cite{Brenier1991}}{brenier}
$X=Y=\mathbb{R}^d$，$c(x,y)=\|x-y\|^2$ とし，
$\mu,\nu\in\mathcal{P}(\mathbb{R}^d)$ が有限の二次モーメントを持つとする．
$\mu$ が Lebesgue 測度に関して絶対連続ならば，次が成り立つ：
\begin{enumerate}
  \item ある凸関数 $\phi\colon\mathbb{R}^d\to\mathbb{R}\cup\{+\infty\}$ が存在して，
        $T=\nabla\phi$ が $\mu$-a.e.\ で定義され，$T_\#\mu=\nu$ を満たす．
  \item この $T$ は Monge 問題の最適輸送写像であり，$\mu$-a.e.\ で一意である．
  \item Monge 問題と Kantorovich 問題の最適値が一致する：
        \[
          \inf_{T:\,T_\#\mu=\nu}\int_{\mathbb{R}^d}\|x-T(x)\|^2\,d\mu(x)
          = \min_{\pi\in\Pi(\mu,\nu)}\int_{\mathbb{R}^d\times\mathbb{R}^d}\|x-y\|^2\,d\pi(x,y).
        \]
\end{enumerate}
\end{theorem}

\begin{example}{1次元ガウス分布間の最適輸送}{gauss_ot}
$X=Y=\mathbb{R}$，$c(x,y)=(x-y)^2$ とし，
$\mu=\mathcal{N}(m_1,\sigma_1^2)$，$\nu=\mathcal{N}(m_2,\sigma_2^2)$
（$\sigma_1,\sigma_2>0$）とする．
$\mu$ は Lebesgue 測度に関して絶対連続であるから，
Brenier の定理（定理~\ref{thm:brenier}）により最適輸送写像が存在する．

1次元では，最適輸送写像は分布関数の逆関数を用いて
$T=F_\nu^{-1}\circ F_\mu$ と表せる．
$\mu,\nu$ がともにガウス分布であるから，
$F_\mu(x)=\Phi\!\left(\frac{x-m_1}{\sigma_1}\right)$，
$F_\nu^{-1}(p)=m_2+\sigma_2\,\Phi^{-1}(p)$
（$\Phi$ は標準正規分布の分布関数）より，
\[
  T(x)=F_\nu^{-1}(F_\mu(x))
  = m_2 + \frac{\sigma_2}{\sigma_1}(x-m_1).
\]
これはアフィン写像（平行移動とスケーリング）であり，
$T_\#\mu=\nu$ が確かに成り立つ
（正規分布のアフィン変換は正規分布）．

最適輸送コストは
\begin{align*}
  W_2(\mu,\nu)^2
  &= \int_{\mathbb{R}} (x-T(x))^2\,d\mu(x)
  = \int_{\mathbb{R}} \left[(m_1-m_2) + \left(1-\tfrac{\sigma_2}{\sigma_1}\right)x
    + \left(\tfrac{\sigma_2}{\sigma_1}-1\right)m_1\right]^2 d\mu(x) \\
  &= (m_1-m_2)^2 + (\sigma_1-\sigma_2)^2
\end{align*}
で与えられる．すなわち，ガウス分布間の 2-Wasserstein 距離は
\[
  \boxed{W_2(\mu,\nu) = \sqrt{(m_1-m_2)^2 + (\sigma_1-\sigma_2)^2}}
\]
となる．平均の差と標準偏差の差のユークリッドノルムという，明快な幾何学的解釈を持つ．

\begin{definition}{輸送計画}{transport_plan}
$X$，$Y$ を位相空間とし，
$\mu$，$\nu$ をそれぞれ $X$，$Y$ 上のBorel確率測度とする．
$X\times Y$ 上のBorel確率測度 $\pi$ が
$\mu$ と $\nu$ の\textbf{輸送計画}であるとは，
$\pi$ の周辺分布がそれぞれ $\mu$ と $\nu$ に一致すること，
すなわち任意のBorel集合 $A\subset X$，$B\subset Y$ に対し
\[
  \pi(A\times Y) = \mu(A), \qquad
  \pi(X\times B) = \nu(B)
\]
が成り立つことをいう．
$\mu$ と $\nu$ の輸送計画全体の集合を $\Pi(\mu,\nu)$ と記す．
\end{definition}

離散的な場合，$\pi$ を行列 $(\pi_{xy})$ で表すと
各成分 $\pi_{xy}$ は「出発地 $x$ にある土を到達地 $y$ へ輸送する量」を意味する．
\begin{itemize}
  \item \textbf{行 $x$ の和}が $\mu(\{x\})$ に一致する条件は，
        「出発地 $x$ から送り出す土の総量がちょうど供給量に等しい」ことを要請する．
  \item \textbf{列 $y$ の和}が $\nu(\{y\})$ に一致する条件は，
        「到達地 $y$ に受け取る土の総量がちょうど需要量に等しい」ことを要請する．
  \item 対角成分 $\pi_{xx}$ はその場に\textbf{留まる}量，
        非対角成分 $\pi_{xy}\ (x\ne y)$ は $x$ から $y$ へ\textbf{移動する}量に対応する．
\end{itemize}

\begin{definition}{Kantorovich 問題}{kantorovich}
$X$，$Y$ を位相空間とし，
$\mu$，$\nu$ をそれぞれ $X$，$Y$ 上のBorel確率測度，
$c\colon X\times Y\to[0,\infty]$ をBorel写像とする．
\textbf{Kantorovich 問題}は
\[
  \inf_{\pi\in\Pi(\mu,\nu)}
  \int_{X\times Y} c(x,y)\,d\pi(x,y)
\]
を求める問題である．
\end{definition}

\begin{remark}{Monge 問題との関係}{monge_kantorovich}
Monge の輸送写像 $T\colon X\to Y$ が与えられたとき，
写像 $x\mapsto(x,\,T(x))$ による $\mu$ の押し出し測度
\[
  \pi(A\times B) = \mu(\{x\in A : T(x)\in B\})
\]
は輸送計画になる．
これは「$x$ を $T(x)$ に確定的に送る」という計画を，
$X\times Y$ 上の同時分布として表現したものである．
したがって Kantorovich 問題は Monge 問題の真の緩和であり，
Kantorovich 問題の最適値は Monge 問題の最適値以下である．
さらに $\Pi(\mu,\nu)$ は常に非空である．
実際，$X\times Y$ 上の測度 $\pi$ を
\[
  \pi(A\times B) \coloneqq \mu(A)\cdot\nu(B)
\]
で定めると，周辺分布条件を満たす：
\begin{align*}
  \pi(A\times Y) &= \mu(A)\cdot\nu(Y) = \mu(A)\cdot 1 = \mu(A), \\
  \pi(X\times B) &= \mu(X)\cdot\nu(B) = 1\cdot\nu(B) = \nu(B).
\end{align*}
よって $\pi\in\Pi(\mu,\nu)$ であり，$\Pi(\mu,\nu)$ は常に非空である．
したがって，Monge 問題とは異なり Kantorovich 問題は常に実行可能である．
\end{remark}

\begin{example}{Kantorovich 問題の具体例（前節の失敗例を解く）}{kantorovich_example}
設定~\ref{set:basic}に対し，Kantorovich 問題を解く．
$X = Y = \{0, 1\}$ とし，
\[
  \mu = \tfrac{3}{4}\delta_0 + \tfrac{1}{4}\delta_1, \qquad
  \nu  = \tfrac{2}{4}\delta_0 + \tfrac{2}{4}\delta_1, \qquad
  c(x,y) = |x-y|
\]
とおく．
2つの分布を棒グラフで示すと，$\mu$ と $\nu$ の確率質量の違いが視覚的にわかる．

\medskip
\textbf{輸送計画の行列表示．}
$X\times Y=\{0,1\}^2$ は有限集合なので，
$\pi$ は4点上の確率質量 $\pi_{xy}\coloneqq\pi(\{x\}\times\{y\})\ge0$ で完全に決まり，
$2\times2$ 行列
\[
  \pi = \begin{pmatrix} \pi_{00} & \pi_{01} \\ \pi_{10} & \pi_{11} \end{pmatrix}
\]
で表せる．輸送計画の定義（定義~\ref{def:transport_plan}）の周辺分布条件
$\pi(A\times Y)=\mu(A)$ および $\pi(X\times B)=\nu(B)$ を，
$A$ と $B$ としてそれぞれ $\{0\}$ と $\{1\}$ に適用すると，
\begin{align}
  \pi(\{0\}\times Y) &= \pi_{00}+\pi_{01} = \mu(\{0\}) = \tfrac{3}{4} \label{eq:marg_mu0}\\
  \pi(\{1\}\times Y) &= \pi_{10}+\pi_{11} = \mu(\{1\}) = \tfrac{1}{4} \label{eq:marg_mu1}\\
  \pi(X\times\{0\}) &= \pi_{00}+\pi_{10} = \nu(\{0\}) = \tfrac{2}{4} \label{eq:marg_nu0}\\
  \pi(X\times\{1\}) &= \pi_{01}+\pi_{11} = \nu(\{1\}) = \tfrac{2}{4} \label{eq:marg_nu1}
\end{align}
すなわち，「行の和が $\mu$，列の和が $\nu$ に一致する非負行列」が
$\Pi(\mu,\nu)$ の元である．

\medskip
\textbf{最適計画の導出．}
$c(x,y)=|x-y|$ を各格子点で評価すると，
輸送コストは
\[
  \int c\,d\pi
  = 0\cdot\pi_{00} + 1\cdot\pi_{01} + 1\cdot\pi_{10} + 0\cdot\pi_{11}
  = \pi_{01} + \pi_{10}
\]
と書ける．\eqref{eq:marg_mu0}，\eqref{eq:marg_nu0} から
$\pi_{01} = \tfrac{3}{4} - \pi_{00}$，$\pi_{10} = \tfrac{2}{4} - \pi_{00}$
なので，コストは
\[
  \pi_{01}+\pi_{10} = \tfrac{5}{4} - 2\pi_{00}
\]
となり，$\pi_{00}$ を最大化することで最小化される．
$\pi_{00}\ge0$ かつ各周辺分布条件から
$\pi_{00}\le\min\!\bigl(\tfrac{3}{4},\,\tfrac{2}{4}\bigr)=\tfrac{2}{4}$ なので，
最適計画は $\pi_{00}=\tfrac{2}{4}$ で与えられ，
\[
  \pi^* = \begin{pmatrix} 2/4 & 1/4 \\ 0 & 1/4 \end{pmatrix},
  \qquad
  \inf_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi
  = \tfrac{5}{4} - \tfrac{4}{4} = \tfrac{1}{4}.
\]

\medskip
$\pi^*$ は $X\times Y=\{0,1\}^2$ 上の同時分布であり，
下の行列の各セルが確率質量 $\pi^*(x,y)$ を表す．
行の和が $\mu$，列の和が $\nu$ に一致することが周辺分布条件に対応する．
オレンジのセルだけが正のコスト（$c=1$）を持ち，
輸送コスト全体への寄与は $\tfrac{1}{4}\times 1=\tfrac{1}{4}$ である．

\begin{definition}{距離空間}{metric_space}
集合 $X$ 上の関数 $d\colon X\times X\to[0,\infty)$ が\textbf{距離}であるとは，
任意の $x,y,z\in X$ に対し次を満たすことをいう：
\begin{enumerate}
  \item $d(x,y)=0 \Leftrightarrow x=y$，
  \item $d(x,y)=d(y,x)$（対称性），
  \item $d(x,z)\le d(x,y)+d(y,z)$（三角不等式）．
\end{enumerate}
組 $(X,d)$ を\textbf{距離空間}という．
\end{definition}

\begin{definition}{Cauchy列と完備性}{complete}
距離空間 $(X,d)$ の点列 $\{x_n\}$ が\textbf{Cauchy列}であるとは，
任意の $\varepsilon>0$ に対し，ある $N$ が存在して
$n,m\ge N$ ならば $d(x_n,x_m)<\varepsilon$ が成り立つことをいう．
距離空間が\textbf{完備}であるとは，任意のCauchy列が $X$ 内の点に収束することをいう．
\end{definition}

Cauchy列は「点同士の距離が0に近づく列」であり，完備性は
「そのような列の行き先が必ず $X$ の中にある」ことを保証する条件である．
不完備な空間では，Cauchy列が空間の「穴」に向かって収束しようとするが，
極限点が $X$ に存在しない．

\begin{definition}{稠密と可分性}{separable}
距離空間 $(X,d)$ の部分集合 $S$ が\textbf{稠密}であるとは，
$X$ の任意の点 $x$ と任意の $\varepsilon>0$ に対し，
$d(x,s)<\varepsilon$ を満たす $s\in S$ が存在することをいう．
距離空間が\textbf{可分}であるとは，稠密な可算部分集合が存在することをいう．
\end{definition}

稠密の直感：$S$ の点が $X$ 全体に「すき間なく散らばっている」ため，
どこを中心にどんなに小さな $\varepsilon$ 球を置いても，必ず $S$ の点が入り込む．

\begin{definition}{下半連続}{lsc}

\[

\]

\end{definition}

\begin{definition}{測度の弱収束}{weak_convergence}
距離空間 $X$ 上のBorel確率測度の列 $\{\mu_n\}$ が
$\mu$ に\textbf{弱収束}するとは，
任意の有界連続関数 $f\colon X\to\mathbb{R}$ に対し
\[
  \lim_{n\to\infty}\int_X f\,d\mu_n = \int_X f\,d\mu
\]
が成り立つことをいう．
\end{definition}

\begin{definition}{緊密性}{tightness}
距離空間 $X$ 上のBorel確率測度の族 $\mathcal{M}$ が\textbf{緊密}（tight）であるとは，
任意の $\varepsilon>0$ に対し，あるコンパクト集合 $K\subset X$ が存在して，
すべての $\mu\in\mathcal{M}$ に対し $\mu(X\setminus K)<\varepsilon$ が成り立つことをいう．
\end{definition}

\begin{theorem}{Prokhorovの定理}{prokhorov}
ポーランド空間上のBorel確率測度の族 $\mathcal{M}$ が緊密であることと，
$\mathcal{M}$ が\textbf{相対コンパクト}
（任意の列が弱収束する部分列を持つ）であることは同値である．
\end{theorem}

\begin{proof}
（相対コンパクト $\Rightarrow$ 緊密）
対偶を示す．$\mathcal{M}$ が緊密でないと仮定すると，
ある $\varepsilon>0$ が存在して，
任意のコンパクト集合 $K$ に対し $\mu_K(X\setminus K)\ge\varepsilon$
を満たす $\mu_K\in\mathcal{M}$ が取れる．
ポーランド空間は $\sigma$コンパクトであるから，
コンパクト集合の増大列 $K_1\subset K_2\subset\cdots$
で $X=\bigcup_n K_n$ となるものが存在する．
各 $K_n$ に対し $\mu_{K_n}(X\setminus K_n)\ge\varepsilon$ を満たす
$\mu_{K_n}\in\mathcal{M}$ を選ぶと，列 $\{\mu_{K_n}\}$ が得られる．

この列が相対コンパクトでないことを示す．
任意の部分列 $\{\mu_{K_{n_k}}\}$ が弱収束先 $\mu$ を持つと仮定する．
各 $K_m$ はコンパクト（したがって閉集合）であるから，
弱収束の Portmanteau 定理（閉集合に関する上半連続性）より
\[
  \mu(X\setminus K_m)
  \ge \limsup_{k\to\infty}\mu_{K_{n_k}}(X\setminus K_m).
\]
$n_k\ge m$ なる $k$ に対しては $K_m\subset K_{n_k}$ であるから
$\mu_{K_{n_k}}(X\setminus K_m)\ge\mu_{K_{n_k}}(X\setminus K_{n_k})\ge\varepsilon$
が成り立ち，$\mu(X\setminus K_m)\ge\varepsilon$ を得る．
$K_m\uparrow X$ であるから $\mu(X)=\lim_{m\to\infty}\mu(K_m)\le 1-\varepsilon<1$
となり，$\mu$ が確率測度であることに矛盾する．
よって相対コンパクトならば緊密である．

\medskip
（緊密 $\Rightarrow$ 相対コンパクト）
$\mathcal{M}$ から任意の列 $\{\mu_n\}$ を取り，
弱収束する部分列の存在を示す．

\textit{ステップ1：可算族上での対角線論法．}
ポーランド空間 $(X,d)$ は可分であるから，
稠密な可算集合 $S=\{q_1,q_2,\ldots\}$ が存在する．
正の有理数全体 $\mathbb{Q}_{>0}$ との直積
$\mathcal{U}\coloneqq\{B(q_k,r) : k\ge 1,\;r\in\mathbb{Q}_{>0}\}$
は可算であり，$X$ の位相の基底をなす．
各 $B\in\mathcal{U}$ に対し，数列 $\{\mu_n(B)\}_{n\ge 1}\subset[0,1]$
は有界であるから，Bolzano--Weierstrass の定理により
収束する部分列が存在する．
$\mathcal{U}$ を $B_1,B_2,\ldots$ と番号付けし，
$B_1$ に対して収束部分列 $\{\mu_{n}^{(1)}\}$ を取り，
$B_2$ に対して $\{\mu_{n}^{(1)}\}$ からさらに収束部分列 $\{\mu_{n}^{(2)}\}$ を取り，
……と繰り返す．
対角部分列 $\mu_{n_k}\coloneqq\mu_{k}^{(k)}$ は
すべての $B\in\mathcal{U}$ に対し $\mu_{n_k}(B)\to\lambda(B)$ を満たす
（何らかの極限 $\lambda(B)\in[0,1]$ が定まる）．

\textit{ステップ2：極限測度の構成と弱収束．}
緊密性により，任意の $\varepsilon>0$ に対しコンパクト集合 $K$ で
$\mu_{n_k}(X\setminus K)<\varepsilon$（$\forall k$）を満たすものが取れる．
これは部分列の「質量の逃散」を防ぎ，
ステップ1の極限 $\lambda$ から $X$ 全体上のBorel確率測度 $\mu$ を
$\mu(G)\coloneqq\sup\{\lambda(B) : B\in\mathcal{U},\;\overline{B}\subset G\}$
（$G$ は開集合）の形で矛盾なく定めることを可能にする．
任意の有界連続関数 $f$ に対し $\int f\,d\mu_{n_k}\to\int f\,d\mu$ が成り立ち，
$\mu_{n_k}$ は $\mu$ に弱収束する．
\end{proof}

\begin{theorem}{Kantorovich問題の最適解の存在}{kantorovich_existence}
$X,Y$ をポーランド空間とし，
$c\colon X\times Y\to[0,\infty]$ が下半連続であるとする．
このとき Kantorovich 問題の下限は最小値として達成される．
すなわち，ある $\pi^*\in\Pi(\mu,\nu)$ が存在して
\[
  \int_{X\times Y} c(x,y)\,d\pi^*(x,y)
  = \inf_{\pi\in\Pi(\mu,\nu)}
    \int_{X\times Y} c(x,y)\,d\pi(x,y)
\]
が成り立つ．
\end{theorem}

\begin{proof}
証明は3つのステップからなる．

\textit{ステップ1：$\Pi(\mu,\nu)$ の緊密性．}
任意の $\varepsilon>0$ に対し，$\mu$ と $\nu$ の緊密性から
コンパクト集合 $K_X\subset X$，$K_Y\subset Y$ で
$\mu(X\setminus K_X)<\varepsilon/2$，$\nu(Y\setminus K_Y)<\varepsilon/2$
を満たすものが存在する．
任意の $\pi\in\Pi(\mu,\nu)$ に対し
\begin{align*}
  \pi\bigl((X\times Y)\setminus(K_X\times K_Y)\bigr)
  &\le \pi((X\setminus K_X)\times Y) + \pi(X\times(Y\setminus K_Y)) \\
  &= \mu(X\setminus K_X) + \nu(Y\setminus K_Y) < \varepsilon.
\end{align*}
$K_X\times K_Y$ は $X\times Y$ のコンパクト集合であるから，
$\Pi(\mu,\nu)$ は緊密である．

\textit{ステップ2：$\Pi(\mu,\nu)$ のコンパクト性．}
ステップ1と Prokhorov の定理（定理~\ref{thm:prokhorov}）から
$\Pi(\mu,\nu)$ は相対コンパクトである．
閉性を示す：$\Pi(\mu,\nu)$ の列 $\pi_n\to\pi$（弱収束）とする．
任意の $f\in C_b(X)$ に対し $g(x,y)\coloneqq f(x)$ は有界連続であるから
\[
  \int_X f\,d(\pi_1)_X
  = \int_{X\times Y} g\,d\pi_n
  \to \int_{X\times Y} g\,d\pi
  = \int_X f\,d\pi_X.
\]
ここで $\pi_X$ は $\pi$ の $X$ への周辺分布を表す．
各 $\pi_n\in\Pi(\mu,\nu)$ より $(\pi_n)_X=\mu$ であるから
$\pi_X=\mu$ を得る．同様に $\pi_Y=\nu$．
よって $\pi\in\Pi(\mu,\nu)$ であり，$\Pi(\mu,\nu)$ は閉である．
相対コンパクトかつ閉であるから $\Pi(\mu,\nu)$ はコンパクトである．

\textit{ステップ3：最小値の達成．}
$c$ が下半連続であることから，
命題~\ref{prop:lsc_approx}と同様の議論により
$\pi\mapsto\int c\,d\pi$ は弱収束に関して下半連続である．
コンパクト集合 $\Pi(\mu,\nu)$ 上の下半連続関数は最小値を達成するから，
ある $\pi^*\in\Pi(\mu,\nu)$ が最適解として存在する．
\end{proof}

\section{双対問題}

Kantorovich 問題は「最小コストの輸送計画を探す」主問題であり，
定理~\ref{thm:kantorovich_existence}によりその最適解は存在する．
本節では，これと双対をなす最大化問題（双対問題）を導入する．
双対問題を解くことで，輸送計画を陽に求めずとも
最適値を確認できる（強双対性，定理~\ref{thm:kantorovich_duality}）．
設定~\ref{set:basic}に対して前節で得た最適コスト $\tfrac{1}{4}$ を，
双対側から検証するのが例~\ref{ex:dual_admissible_example} である．

\begin{definition}{双対許容対}{dual_admissible}
コスト関数 $c\colon X\times Y\to[0,\infty]$ に対し，
関数対 $(\varphi,\psi)\in L^1(\mu)\times L^1(\nu)$ が
\[
  \varphi(x)+\psi(y)\le c(x,y)
  \quad \text{($\mu$-a.e. $x$，$\nu$-a.e. $y$)}
\]
を満たすとき，$(\varphi,\psi)$ を\textbf{双対許容対}（dual admissible pair）という．
双対許容対全体の集合を
\[
  \Phi_c = \bigl\{(\varphi,\psi)\in L^1(\mu)\times L^1(\nu)
  \mid \varphi(x)+\psi(y)\le c(x,y)\bigr\}
\]
と表す．
\end{definition}

$\varphi(x)$ を「出発地 $x$ での積み込み手数料」，$\psi(y)$ を「到着地 $y$ での荷降ろし手数料」と解釈すると，
$\varphi(x)+\psi(y)$ は $x\to y$ の輸送で仲介業者が受け取る\textbf{合計手数料}である．
制約 $\varphi(x)+\psi(y)\le c(x,y)$ は，
「仲介業者の合計手数料が直接輸送コストを超えない」ことを要求する——
もし超えるなら荷主は仲介業者を使わず自前で運んだほうが安いため，合理的な市場では成立しない条件である．
双対問題はこの制約のもとで総収入 $\int_X\varphi\,d\mu+\int_Y\psi\,d\nu$ を最大化する手数料設定を求める問題である．

なお，$L^1(\mu)$ とは $\mu$ に関して可積分な関数のクラス，
すなわち $\int_X|f|\,d\mu < \infty$ を満たす関数 $f$ 全体を指す．
$(\varphi,\psi)\in L^1(\mu)\times L^1(\nu)$ という条件は，
総収入の積分 $\int\varphi\,d\mu$，$\int\psi\,d\nu$ が有限値として定まることを保証するための技術的な仮定である．

\begin{example}{双対許容対の具体例（設定~\ref{set:basic}での検証）}{dual_admissible_example}
設定~\ref{set:basic}に対して双対問題を考える：
$X=Y=\{0,1\}$，$\mu=\tfrac{3}{4}\delta_0+\tfrac{1}{4}\delta_1$，
$\nu=\tfrac{2}{4}\delta_0+\tfrac{2}{4}\delta_1$，$c(x,y)=|x-y|$．
例~\ref{ex:kantorovich_example} で求めた最適輸送コスト $\tfrac{1}{4}$ を双対側から確認する．

双対許容対の条件 $\varphi(x)+\psi(y)\le c(x,y)$ を $(x,y)$ の各組み合わせで書くと，
\begin{align*}
  \varphi(0)+\psi(0) &\le c(0,0) = 0, \\
  \varphi(0)+\psi(1) &\le c(0,1) = 1, \\
  \varphi(1)+\psi(0) &\le c(1,0) = 1, \\
  \varphi(1)+\psi(1) &\le c(1,1) = 0.
\end{align*}
双対問題が最大化する\textbf{総収入}とは，$\int_X\varphi\,d\mu+\int_Y\psi\,d\nu$ のことである．
離散分布の場合，積分はそれぞれの点への質量の重み付き和になるので，
\[
  \int_X\varphi\,d\mu+\int_Y\psi\,d\nu
  = \varphi(0)\cdot\mu(\{0\})+\varphi(1)\cdot\mu(\{1\})
    +\psi(0)\cdot\nu(\{0\})+\psi(1)\cdot\nu(\{1\}).
\]
これは「各地点の手数料 × その地点を通る質量」の総和，すなわち仲介業者の総取り分である．

例えば $\varphi\equiv 0,\,\psi\equiv 0$ は制約を満たすが，
総収入は $0\cdot\tfrac{3}{4}+0\cdot\tfrac{1}{4}+0\cdot\tfrac{2}{4}+0\cdot\tfrac{2}{4}=0$ である．

一方，$\varphi(0)=0,\,\varphi(1)=-1,\,\psi(0)=0,\,\psi(1)=1$ とすると，
4つの制約は下の表のように全て成立する（緑＝等号，青＝余裕あり）．

\begin{theorem}{Kantorovich 双対定理}{kantorovich_duality}
$X,Y$ をポーランド空間とし，
$c\colon X\times Y\to[0,\infty]$ が下半連続とする．このとき
\[
  \inf_{\pi\in\Pi(\mu,\nu)}
  \int_{X\times Y} c(x,y)\,d\pi(x,y)
  = \sup_{(\varphi,\psi)\in\Phi_c}
  \left(
    \int_{X}\varphi(x)\,d\mu(x)
    + \int_{Y}\psi(y)\,d\nu(y)
  \right)
\]
が成り立つ．
\end{theorem}

以下では定理~\ref{thm:kantorovich_duality}を証明する．
方針は次の通りである：
(1) 弱双対性（$\ge$方向）を直接計算で示し，
(2) コンパクトな場合にミニマックス定理を用いて強双対性を証明し，
(3) 一般のポーランド空間・下半連続コストへ拡張する．

\begin{proposition}{弱双対性}{weak_duality}
任意の $\pi\in\Pi(\mu,\nu)$ と $(\varphi,\psi)\in\Phi_c$ に対し
\[
  \int_X\varphi\,d\mu + \int_Y\psi\,d\nu
  \le \int_{X\times Y} c(x,y)\,d\pi(x,y)
\]
が成り立つ．
\end{proposition}

\begin{proof}
$\pi$ の周辺分布が $\mu$，$\nu$ であることから
\[
  \int_X\varphi\,d\mu + \int_Y\psi\,d\nu
  = \int_{X\times Y}\bigl[\varphi(x)+\psi(y)\bigr]\,d\pi(x,y).
\]
双対許容条件 $\varphi(x)+\psi(y)\le c(x,y)$ を被積分関数に適用すると
\[
  \int_{X\times Y}\bigl[\varphi(x)+\psi(y)\bigr]\,d\pi(x,y)
  \le \int_{X\times Y} c(x,y)\,d\pi(x,y).
\]
\end{proof}

弱双対性から直ちに
$\sup_{(\varphi,\psi)\in\Phi_c}(\int\varphi\,d\mu+\int\psi\,d\nu)
\le\inf_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi$
が従う．残る課題は逆向きの不等式（強双対性）の証明である．

\begin{proposition}{下半連続関数の単調近似}{lsc_approx}
距離空間 $(Z,d)$ 上の下半連続関数 $f\colon Z\to[0,\infty]$ に対し，
有界連続関数の非減少列 $\{f_k\}_{k=1}^\infty$ で
$0\le f_1\le f_2\le\cdots$ かつ $f_k(z)\uparrow f(z)$（各点収束）
を満たすものが存在する．
\end{proposition}

\begin{proof}
$f_k(z)\coloneqq\min\bigl(\inf_{w\in Z}[f(w)+k\,d(z,w)],\;k\bigr)$
と定めれば，各 $f_k$ は $k$-Lipschitz 連続かつ $0\le f_k\le k$ で有界であり，
$f_k\le f_{k+1}\le f$ は定義から明らかである．
$f_k(z)\uparrow f(z)$ は $f$ の下半連続性から従う：
任意の $\varepsilon>0$ に対し $f$ の下半連続性より
$z$ の近傍で $f(w)>f(z)-\varepsilon$ が成り立ち，
$k$ が十分大きければ $\inf_w[f(w)+k\,d(z,w)]\ge f(z)-\varepsilon$ を得る．
\end{proof}

強双対性の証明には，次のミニマックス定理を用いる．

\begin{theorem}{ミニマックス定理（Fan~\cite{Fan1953}）}{minimax}
$V$ を局所凸位相ベクトル空間，$K\subset V$ をコンパクト凸集合，
$L$ をベクトル空間の凸集合とする．
関数 $F\colon K\times L\to\mathbb{R}$ が
\begin{itemize}
  \item 各 $\ell\in L$ に対し $k\mapsto F(k,\ell)$ が凸かつ下半連続，
  \item 各 $k\in K$ に対し $\ell\mapsto F(k,\ell)$ が凹
\end{itemize}
であるとき，
\[
  \min_{k\in K}\sup_{\ell\in L} F(k,\ell)
  = \sup_{\ell\in L}\min_{k\in K} F(k,\ell)
\]
が成り立つ（左辺の $\inf$ は $\min$ として達成される）．
\end{theorem}

\begin{proposition}{連続コストに対する双対性（コンパクトな場合）}{duality_compact}
$X,Y$ を\textbf{コンパクトな}ポーランド空間，$c\in C(X\times Y)$，$c\ge 0$ とする．このとき
\[
  \min_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi
  = \sup_{\substack{\varphi\in C(X),\;\psi\in C(Y)\\
    \varphi(x)+\psi(y)\le c(x,y)}}
  \left(\int_X\varphi\,d\mu+\int_Y\psi\,d\nu\right).
\]
\end{proposition}

\begin{proof}
$X\times Y$ がコンパクトであるから，
Borel確率測度の全体 $\mathcal{P}(X\times Y)$ は弱収束に関してコンパクトな凸集合である．
ミニマックス定理（定理~\ref{thm:minimax}）を
$K=\mathcal{P}(X\times Y)$，$L=C(X)\times C(Y)$ に対し，関数
\[
  \Psi(\gamma,(\varphi,\psi))
  \coloneqq \int_{X\times Y}\bigl[c(x,y)-\varphi(x)-\psi(y)\bigr]\,d\gamma
    + \int_X\varphi\,d\mu + \int_Y\psi\,d\nu
\]
に適用する．$\Psi$ は $\gamma$ に関して線形（凸）かつ弱収束に関して連続であり，
$(\varphi,\psi)$ に関して線形（凹）である．よって
\[
  \min_{\gamma\in\mathcal{P}(X\times Y)}
  \sup_{(\varphi,\psi)\in L}\Psi
  = \sup_{(\varphi,\psi)\in L}
  \min_{\gamma\in\mathcal{P}(X\times Y)}\Psi.
\]

\textbf{左辺の計算．}
$\gamma$ の周辺分布を $\gamma_1=(p_1)_\#\gamma$，$\gamma_2=(p_2)_\#\gamma$ と書くと
\[
  \Psi(\gamma,(\varphi,\psi))
  = \int c\,d\gamma
    + \int\varphi\,d(\mu-\gamma_1)
    + \int\psi\,d(\nu-\gamma_2).
\]
$\gamma\in\Pi(\mu,\nu)$（すなわち $\gamma_1=\mu$，$\gamma_2=\nu$）のとき
$\sup_{(\varphi,\psi)}\Psi=\int c\,d\gamma$ である．
$\gamma\notin\Pi(\mu,\nu)$ のとき，例えば $\gamma_1\neq\mu$ ならば
$X$ がコンパクトであるから $\int\varphi\,d(\mu-\gamma_1)\neq 0$
を満たす $\varphi\in C(X)$ が存在し，
スカラー倍により $\sup_{(\varphi,\psi)}\Psi=+\infty$ となる．
よって
\[
  \text{左辺}
  = \min_{\gamma\in\Pi(\mu,\nu)}\int c\,d\gamma.
\]

\textbf{右辺の計算．}
固定した $(\varphi,\psi)$ に対し，$\gamma$ に関して $\Psi$ を最小化する．
$\Psi(\gamma,(\varphi,\psi))=\int[c-\varphi\oplus\psi]\,d\gamma + (\text{定数})$
であるから，Dirac測度 $\gamma=\delta_{(x_0,y_0)}$ を
$c(x,y)-\varphi(x)-\psi(y)$ の最小点（$X\times Y$ のコンパクト性で存在）に置けば
\[
  \min_\gamma\Psi
  = \min_{(x,y)\in X\times Y}\bigl[c(x,y)-\varphi(x)-\psi(y)\bigr]
    + \int\varphi\,d\mu + \int\psi\,d\nu.
\]
任意の $(\varphi,\psi)\in C(X)\times C(Y)$ に対し
$m\coloneqq\min_{(x,y)}[c(x,y)-\varphi(x)-\psi(y)]$ とおき
$\varphi'=\varphi+m$，$\psi'=\psi$ と定めると
$\varphi'(x)+\psi'(y)\le c(x,y)$ が成り立ち，かつ
\[
  \int\varphi'\,d\mu+\int\psi'\,d\nu
  = \int\varphi\,d\mu+\int\psi\,d\nu + m
  = \min_\gamma\Psi(\gamma,(\varphi,\psi)).
\]
逆に $\varphi+\psi\le c$ ならば $m\ge 0$ なので
$\min_\gamma\Psi\ge\int\varphi\,d\mu+\int\psi\,d\nu$．以上から
\[
  \text{右辺}
  = \sup_{\substack{\varphi\in C(X),\;\psi\in C(Y)\\
    \varphi(x)+\psi(y)\le c(x,y)}}
  \Bigl(\int\varphi\,d\mu+\int\psi\,d\nu\Bigr).
\]
ミニマックス等式の左辺＝右辺から所望の等式を得る．
\end{proof}

\begin{proposition}{有界連続コストに対する双対性}{duality_bounded}
$X,Y$ をポーランド空間，$\mu\in\mathcal{P}(X)$，$\nu\in\mathcal{P}(Y)$，
$c\in C_b(X\times Y)$，$c\ge 0$ とする．このとき
\[
  \min_{\pi\in\Pi(\mu,\nu)}\int_{X\times Y} c\,d\pi
  = \sup_{\substack{(\varphi,\psi)\in L^1(\mu)\times L^1(\nu)\\
    \varphi(x)+\psi(y)\le c(x,y)}}
  \left(\int_X\varphi\,d\mu+\int_Y\psi\,d\nu\right).
\]
\end{proposition}

\begin{proof}
$M\coloneqq\|c\|_\infty$ とおく．
弱双対性（命題~\ref{prop:weak_duality}）から $P\ge D$ は既知である．
逆の不等式 $P\le D$ を示す．

\textbf{ステップ1：コンパクト集合による内側近似．}
ポーランド空間上のBorel確率測度は緊密であるから，
コンパクト集合の増大列 $K_1\subset K_2\subset\cdots\subset X$，
$L_1\subset L_2\subset\cdots\subset Y$ で
\[
  \mu(K_n)\to 1,\quad \nu(L_n)\to 1
  \quad (n\to\infty)
\]
を満たすものが存在する．$n$ が十分大きいとき $\mu(K_n)>0$，$\nu(L_n)>0$ であるから，
以下そのような $n$ のみを考える．
$K_n,L_n$ 上の制限測度
\[
  \mu_n\coloneqq\frac{\mu(\,\cdot\,\cap K_n)}{\mu(K_n)},\quad
  \nu_n\coloneqq\frac{\nu(\,\cdot\,\cap L_n)}{\nu(L_n)}
\]
を定めると，$\mu_n\in\mathcal{P}(K_n)$，$\nu_n\in\mathcal{P}(L_n)$ であり，
任意の $f\in C_b(X)$ に対し
\[
  \int f\,d\mu_n = \frac{1}{\mu(K_n)}\int_{K_n}f\,d\mu
  \;\to\; \int_X f\,d\mu
  \quad(n\to\infty)
\]
が成り立つ（有界収束定理と $\mu(K_n)\to 1$ による）．
同様に $\nu_n\to\nu$ も弱収束する．

\textbf{ステップ2：コンパクト空間上の双対性の適用．}
$K_n,L_n$ はコンパクトなポーランド空間であり，
$c|_{K_n\times L_n}$ は連続かつ非負であるから，
命題~\ref{prop:duality_compact}により
\[
  P_n \coloneqq \min_{\pi\in\Pi(\mu_n,\nu_n)}
    \int_{K_n\times L_n} c\,d\pi
  = \sup_{\substack{\varphi\in C(K_n),\;\psi\in C(L_n)\\
    \varphi(x)+\psi(y)\le c(x,y)}}
    \left(\int_{K_n}\varphi\,d\mu_n+\int_{L_n}\psi\,d\nu_n\right)
  \eqqcolon D_n.
\]
各 $n$ に対し，$D_n-\frac{1}{n}$ 以上の双対値を達成する
$(\varphi_n,\psi_n)\in C(K_n)\times C(L_n)$ で
$\varphi_n(x)+\psi_n(y)\le c(x,y)$
（$\forall (x,y)\in K_n\times L_n$）を満たすものを取る．
$c\ge 0$ と $c\le M$ から，$\varphi_n,\psi_n$ を
$[-M,\,M]$ に切り詰めても許容性は保たれるので，
$|\varphi_n|\le M$，$|\psi_n|\le M$ と仮定してよい．

\textbf{ステップ3：双対変数の拡張．}
$X$ および $Y$ 全体上の関数を
\[
  \tilde{\varphi}_n(x)\coloneqq
  \begin{cases}
    \varphi_n(x) & (x\in K_n),\\
    -M & (x\notin K_n),
  \end{cases}
  \qquad
  \tilde{\psi}_n(y)\coloneqq
  \begin{cases}
    \psi_n(y) & (y\in L_n),\\
    -M & (y\notin L_n)
  \end{cases}
\]
で定める．任意の $(x,y)\in X\times Y$ に対する許容性を確認する：
\begin{itemize}
  \item $x\in K_n,\;y\in L_n$：
    $\tilde{\varphi}_n(x)+\tilde{\psi}_n(y)
    =\varphi_n(x)+\psi_n(y)\le c(x,y)$．
  \item $x\notin K_n,\;y\in L_n$：
    $\tilde{\varphi}_n(x)+\tilde{\psi}_n(y)
    =-M+\psi_n(y)\le -M+M=0\le c(x,y)$．
  \item $x\in K_n,\;y\notin L_n$：
    $\tilde{\varphi}_n(x)+\tilde{\psi}_n(y)
    =\varphi_n(x)+(-M)\le M-M=0\le c(x,y)$．
  \item $x\notin K_n,\;y\notin L_n$：
    $\tilde{\varphi}_n(x)+\tilde{\psi}_n(y)=-2M\le 0\le c(x,y)$．
\end{itemize}
よって $(\tilde{\varphi}_n,\tilde{\psi}_n)\in\Phi_c$ であり，
\begin{align*}
  D
  &\ge \int_X\tilde{\varphi}_n\,d\mu+\int_Y\tilde{\psi}_n\,d\nu \\
  &= \mu(K_n)\int_{K_n}\varphi_n\,d\mu_n
    - M\bigl(1-\mu(K_n)\bigr)
    + \nu(L_n)\int_{L_n}\psi_n\,d\nu_n
    - M\bigl(1-\nu(L_n)\bigr).
\end{align*}
$a_n\coloneqq\int\varphi_n\,d\mu_n$，$b_n\coloneqq\int\psi_n\,d\nu_n$
とおくと $a_n+b_n\ge D_n-\frac{1}{n}$ かつ $|a_n|,|b_n|\le M$ であり，
\begin{align*}
  D
  &\ge \mu(K_n)\,a_n + \nu(L_n)\,b_n - M\bigl(2-\mu(K_n)-\nu(L_n)\bigr) \\
  &= (a_n+b_n)
    + \bigl(\mu(K_n)-1\bigr)a_n
    + \bigl(\nu(L_n)-1\bigr)b_n
    - M\bigl(2-\mu(K_n)-\nu(L_n)\bigr) \\
  &\ge D_n - \tfrac{1}{n}
    - 2M\bigl(1-\mu(K_n)\bigr) - 2M\bigl(1-\nu(L_n)\bigr).
\end{align*}
$n\to\infty$ で右辺の誤差項はすべて $0$ に収束するから，
\[
  D \ge \liminf_{n\to\infty} D_n.
\]

\textbf{ステップ4：主問題の値の下界．}
$D_n=P_n$ であるから，$\liminf P_n\ge P$ を示せば証明が完成する．
各 $P_n$ の最適計画を $\pi_n^*\in\Pi(\mu_n,\nu_n)$ とする．
$\pi_n^*$ は $X\times Y$ 上のBorel確率測度とみなせる（$K_n\times L_n$ の外で $0$）．
$\mu_n\to\mu$，$\nu_n\to\nu$（弱収束）であるから
$\{\pi_n^*\}$ は緊密であり，
Prokhorov の定理（定理~\ref{thm:prokhorov}）から
弱収束する部分列 $\pi_{n_k}^*\to\hat{\pi}$ が取れる．
周辺分布条件は弱収束で保たれるから $\hat{\pi}\in\Pi(\mu,\nu)$ であり，
$c$ が有界連続であることから
\[
  P \le \int c\,d\hat{\pi}
  = \lim_{k\to\infty}\int c\,d\pi_{n_k}^*
  = \lim_{k\to\infty} P_{n_k}.
\]
よって $P\le\liminf_{n\to\infty}P_n$ が成り立つ．

\textbf{結論．}
ステップ3--4を合わせると
\[
  D \ge \liminf_{n\to\infty} D_n
  = \liminf_{n\to\infty} P_n
  \ge P.
\]
弱双対性 $P\ge D$ と合わせて $P=D$ を得る．
\end{proof}

命題~\ref{prop:duality_bounded}を用いて，定理~\ref{thm:kantorovich_duality}を証明する．

\begin{proof}[定理~\ref{thm:kantorovich_duality}の証明]
$(\ge)$：弱双対性（命題~\ref{prop:weak_duality}）から直ちに従う．

$(\le)$：命題~\ref{prop:lsc_approx}により，
$0\le c_1\le c_2\le\cdots$ かつ $c_k\uparrow c$（各点収束）を満たす
有界連続関数の列 $\{c_k\}\subset C_b(X\times Y)$ が存在する．
各 $c_k$ に対し，命題~\ref{prop:duality_bounded}から
\[
  P_k\coloneqq\min_{\pi\in\Pi(\mu,\nu)}\int c_k\,d\pi
  = \sup_{(\varphi,\psi)\in\Phi_{c_k}}
  \Bigl(\int\varphi\,d\mu+\int\psi\,d\nu\Bigr)
  \eqqcolon D_k
\]
が成り立つ．

\textbf{$P_k\uparrow P$ の証明．}
$c_k\le c$ から $P_k\le P\coloneqq\min_\pi\int c\,d\pi$ は明らかである．
逆向きの評価のため，各 $P_k$ の最適計画を $\pi_k^*$ とする．
$\Pi(\mu,\nu)$ のコンパクト性から部分列（同じ記号で書く）で
$\pi_k^*\to\hat{\pi}\in\Pi(\mu,\nu)$（弱収束）となるものが取れる．
任意の固定した $m$ に対し，$k\ge m$ のとき $c_m\le c_k$ であるから
\[
  \int c_m\,d\hat{\pi}
  = \lim_{k\to\infty}\int c_m\,d\pi_k^*
  \le \liminf_{k\to\infty}\int c_k\,d\pi_k^*
  = \liminf_{k\to\infty} P_k.
\]
$m\to\infty$ で単調収束定理から $\int c\,d\hat{\pi}\le\liminf P_k$ を得る．
$P\le\int c\,d\hat{\pi}$ であるから
$P\le\liminf P_k\le\limsup P_k\le P$，
よって $P_k\uparrow P$．

\textbf{双対性の完成．}
$c_k\le c$ から $\Phi_{c_k}\subset\Phi_c$，したがって
$D_k\le D\coloneqq\sup_{\Phi_c}(\int\varphi\,d\mu+\int\psi\,d\nu)$ である．
一方 $P_k=D_k$ であったから
\[
  P = \lim_{k\to\infty} P_k = \lim_{k\to\infty} D_k \le D.
\]
弱双対性 $P\ge D$ と合わせて $P=D$ を得る．
\end{proof}

双対問題には明快な経済的解釈がある．
$\varphi(x)$ を出発地 $x$ での積み込み価格，
$\psi(y)$ を到着地 $y$ での荷降ろし価格と考えると，
制約 $\varphi(x)+\psi(y)\le c(x,y)$ は
裁定条件（価格差が輸送コストを超えない）を意味する．
双対問題は，この裁定条件の下で仲介業者の総収入を最大化する問題であり，
強双対性は最適輸送コストと最適価格設定による収入が一致することを述べている．

\begin{example}{Kantorovich 双対定理の検証（設定~\ref{set:basic}）}{duality_verification}
設定~\ref{set:basic}において Kantorovich 双対定理（定理~\ref{thm:kantorovich_duality}）を検証する．
例~\ref{ex:dual_admissible_example} では一組の双対許容対で総収入 $\tfrac{1}{4}$ を示したが，
ここではその対が\textbf{最適}であることを証明し，
相補性条件と $c$変換との整合性を確認する．

\medskip
\textbf{Step 1：最適双対対の導出．}\par\nobreak
双対変数 $\varphi(0),\,\varphi(1),\,\psi(0),\,\psi(1)$ に対する双対問題は
\[
  \max\;\tfrac{3}{4}\varphi(0)+\tfrac{1}{4}\varphi(1)
  +\tfrac{2}{4}\psi(0)+\tfrac{2}{4}\psi(1)
  \quad\text{s.t.}\quad \varphi(x)+\psi(y)\le c(x,y)\;\;(\forall\,x,y)
\]
と書ける．$c(0,0)=c(1,1)=0$ から
$\psi(0)\le -\varphi(0)$，$\psi(1)\le -\varphi(1)$ であり，
目的関数の最大化には等号
$\psi(0)=-\varphi(0)$，$\psi(1)=-\varphi(1)$
で取るのが最適である．代入すると目的関数は $\tfrac{1}{4}(\varphi(0)-\varphi(1))$ となり，
$c(0,1)=c(1,0)=1$ からの制約 $|\varphi(0)-\varphi(1)|\le 1$ の下で最大化すると
$\varphi(0)-\varphi(1)=1$ が最適となる．
$\varphi(0)=0$ と正規化して
\[
  \boxed{\varphi^*(0)=0,\quad \varphi^*(1)=-1,\quad
  \psi^*(0)=0,\quad \psi^*(1)=1}
\]
を得る．最適双対値は $\tfrac{1}{4}$ であり，
例~\ref{ex:kantorovich_example} の最適輸送コスト $\tfrac{1}{4}$ に一致する：
\[
  \underbrace{\inf_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi}_{=\,1/4}
  \;=\;
  \underbrace{\sup_{(\varphi,\psi)\in\Phi_c}
  \Bigl(\int\varphi\,d\mu+\int\psi\,d\nu\Bigr)}_{=\,1/4}.
\]

\medskip
\textbf{Step 2：相補性条件の確認．}\par\nobreak
主問題の最適計画
$\pi^*=\bigl(\begin{smallmatrix}2/4 & 1/4\\0 & 1/4\end{smallmatrix}\bigr)$
と最適双対対 $(\varphi^*,\psi^*)$ の間には，\textbf{相補性条件}（complementary slackness）
\[
  \pi^*_{xy}>0 \implies \varphi^*(x)+\psi^*(y)=c(x,y)
\]
が成り立つはずである（LP の強双対性の帰結）．各 $(x,y)$ で確認する：

という経済的直観と整合している．

  \varphi^*(x)+\psi^*(y) < c(x,y) \implies \pi^*_{xy}=0.
\]

\medskip
\textbf{Step 3：$c$変換との整合性．}\par\nobreak
定義~\ref{def:c_transform} の $c$変換
$\varphi^c(y)=\inf_{x\in X}[c(x,y)-\varphi(x)]$
を最適な $\varphi^*$ に対して計算する：
\begin{align*}
  (\varphi^*)^c(0)
  &= \min\bigl\{c(0,0)-\varphi^*(0),\;\; c(1,0)-\varphi^*(1)\bigr\}
  = \min\{0-0,\;\; 1-(-1)\} = \min\{0,2\} = 0, \\
  (\varphi^*)^c(1)
  &= \min\bigl\{c(0,1)-\varphi^*(0),\;\; c(1,1)-\varphi^*(1)\bigr\}
  = \min\{1-0,\;\; 0-(-1)\} = \min\{1,1\} = 1.
\end{align*}
したがって $(\varphi^*)^c(0)=0=\psi^*(0)$，$(\varphi^*)^c(1)=1=\psi^*(1)$，すなわち
\[
  \boxed{\psi^* = (\varphi^*)^c}
\]
が成り立つ．最適な双対対では $\psi$ が $\varphi$ の $c$変換で与えられるという，
双対理論の一般的結果がこの具体例で確認できた．

さらに $(\varphi^*)^c$ の各 $y$ での $\min$ を達成する $x$ を見ると：
\begin{itemize}
  \item $y=0$：$\min$ を達成するのは $x=0$（$c(0,0)-\varphi^*(0)=0$）．
  \item $y=1$：$\min$ を達成するのは $x=0$ と $x=1$ の両方（ともに値 $1$）．
\end{itemize}
最適計画 $\pi^*$ で質量が乗る $(x,y)$ の組——$(0,0)$, $(0,1)$, $(1,1)$——は
まさに $c$変換の $\inf$ が達成される $(x,y)$ の組に一致しており，
これは「最適計画は $c$変換の $\inf$ を達成する点上に集中する」という一般原理の具現化である．
\end{example}

\begin{remark}{Monge 問題への回帰}{monge_from_kantorovich}
Kantorovich 問題の最適計画 $\pi^*$ が
$\pi^*=({\rm id},T)_\#\mu$ の形に書ける（すなわちグラフ上に集中する）場合，
$T$ は Monge 問題の最適写像でもある．
例えば $X=Y=\mathbb{R}^d$ で $c(x,y)=\|x-y\|^2$ かつ $\mu$ が絶対連続のとき，
Brenier の定理により最適輸送写像 $T=\nabla\phi$
（凸関数の勾配）の存在と一意性が保証される．
\end{remark}

\begin{remark}{Kantorovich 問題と凸解析}{kantorovich_convex}
Kantorovich 問題
\[
  \inf_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi
\]
は，その定義の時点で既に\textbf{無限次元線形計画問題}である：
目的関数 $\int c\,d\pi$ は $\pi$ に関して線形であり，
実行可能領域 $\Pi(\mu,\nu)$ は周辺分布条件という線形制約で定まる凸集合である．

つまり Monge 問題を Kantorovich 問題へ緩和する操作は，
「写像から測度へ変数を広げる」だけでなく，
\textbf{最適輸送を凸解析の問題に埋め込む}ことを意味していた．
Kantorovich 双対定理（定理~\ref{thm:kantorovich_duality}）は
この凸性の帰結として，線形計画の双対定理の無限次元版として導かれる．
有限離散分布の場合は有限次元の LP に帰着し，
単体法で最適輸送計画と最適双対対を同時に求められる（第4節）．
その双対構造を記述する中心的な道具が，次に定義する $c$変換である．
\end{remark}

双対問題は関数の対 $(\varphi,\psi)$ にわたる最適化であるが，
制約 $\varphi(x)+\psi(y)\le c(x,y)$ から
$\psi(y)$ の最良の選択は $\varphi$ で決まる——
すなわち $\psi(y)=\inf_x[c(x,y)-\varphi(x)]$ と取ればよい．
この操作を $\varphi$ の $c$変換と呼ぶ．

\begin{definition}{$c$変換と$c$凹関数}{c_transform}
コスト関数 $c\colon X\times Y\to[0,\infty]$ に対し，
関数 $\varphi\colon X\to\mathbb{R}\cup\{-\infty\}$ の
\textbf{$c$変換}（$c$-transform）を
\[
  \varphi^c(y) \coloneqq \inf_{x\in X}\bigl[c(x,y)-\varphi(x)\bigr]
\]
と定義する．$\varphi = (\varphi^c)^c$ が成り立つとき，
$\varphi$ を\textbf{$c$凹}（$c$-concave）であるという．
\end{definition}

$c$変換を用いると，双対問題の最適化変数を一つの関数に絞ることができる．
任意の双対許容対 $(\varphi,\psi)\in\Phi_c$ に対し，
定義から $\psi(y)\le c(x,y)-\varphi(x)$ が任意の $x$ で成り立つため
$\psi(y)\le\varphi^c(y)$，したがって $\int\psi\,d\nu\le\int\varphi^c\,d\nu$．
$\psi$ を $\varphi^c$ で置き換えても目的関数は悪化しないから，
\[
  \sup_{(\varphi,\psi)\in\Phi_c}
  \Bigl(\int_X\varphi\,d\mu + \int_Y\psi\,d\nu\Bigr)
  = \sup_{\varphi\in L^1(\mu)}
  \Bigl(\int_X\varphi\,d\mu + \int_Y\varphi^c\,d\nu\Bigr).
\]
すなわち双対問題は，関数の対 $(\varphi,\psi)$ の最適化から
一つの関数 $\varphi$ の最適化へと帰着する．
これは凸解析における Legendre--Fenchel 変換が共役対 $(f,\,f^*)$ の最適化を
$f$ 単体の最適化に帰着させるのと同じ仕組みである．

\begin{remark}{Legendre--Fenchel 変換との類比}{legendre_fenchel}
古典的な\textbf{Legendre--Fenchel 変換}（凸共役）は
\[
  f^*(y) \coloneqq \sup_{x\in X}\bigl[\langle x,y\rangle - f(x)\bigr]
\]
で定義される（$X=Y=\mathbb{R}^d$）．
$c$変換は，双線型ペアリング $\langle x,y\rangle$ を一般のコスト $c(x,y)$ で置き換え，
$\sup$ を $\inf$ に変えた「一般化 Legendre--Fenchel 変換」と見ることができる．

この対応を並べると，最適輸送理論は凸解析と平行な構造を持つことが見えてくる：

\medskip
\begin{center}
{\renewcommand{\arraystretch}{1.4}
\begin{tabular}{ll}
  \hline
  \textbf{通常の凸解析} & \textbf{$c$凸解析（最適輸送）} \\
  \hline
  凸関数 & $c$凹関数 \\
  Legendre--Fenchel 変換 $f^*$ & $c$変換 $\varphi^c$ \\
  Fenchel の双対定理 & Kantorovich 双対定理 \\
  \hline
\end{tabular}}
\end{center}
\medskip

特に $c(x,y)=\tfrac{1}{2}\|x-y\|^2$ のとき，
$c$凹関数 $\varphi$ は $\varphi(x)=\tfrac{1}{2}\|x\|^2-u(x)$
（$u$ は凸関数）の形で書け，$\varphi$ の $c$変換は
$u$ の Legendre--Fenchel 共役 $u^*$ を通じて $\varphi^c(y)=\tfrac{1}{2}\|y\|^2-u^*(y)$
と表される．Brenier の定理（注意~\ref{rem:monge_from_kantorovich}）が述べる
凸ポテンシャル $T=\nabla u$ は，この $c$凸解析の構造から自然に現れる．
\end{remark}

\section{離散最適輸送の計算：線形計画法}

有限離散分布に対する Kantorovich 問題は
有限次元の\textbf{線形計画問題}（LP）に帰着する．
注意~\ref{rem:kantorovich_convex}で見たように
Kantorovich 問題は本質的に凸最適化であるが，
離散の場合はさらに\textbf{線形}最適化になり，
単体法という効率的なアルゴリズムで系統的に解くことができる．
本節では LP としての構造を明示し，
輸送問題に特化した\textbf{単体法}（simplex method）によって
設定~\ref{set:basic}を解く過程を追う．

\begin{definition}{線形計画問題（標準形）}{lp}
$c\in\mathbb{R}^n$，$A\in\mathbb{R}^{m\times n}$，$b\in\mathbb{R}^m$ とする．
\textbf{線形計画問題}（LP）の標準形とは
\[
  \min_{x\in\mathbb{R}^n}\; c^\top x
  \quad\text{s.t.}\quad Ax=b,\;\; x\ge 0
\]
をいう．
実行可能集合 $\{x\in\mathbb{R}^n : Ax=b,\;x\ge 0\}$ は
\textbf{凸多面体}（convex polyhedron）をなす．
\end{definition}

\begin{remark}{離散 Kantorovich 問題は LP である}{kantorovich_lp}
有限集合 $X=\{x_1,\ldots,x_m\}$，$Y=\{y_1,\ldots,y_n\}$ 上の
離散確率測度 $\mu$，$\nu$ に対し，
$\mu_i\coloneqq\mu(\{x_i\})$，$\nu_j\coloneqq\nu(\{y_j\})$，
$c_{ij}\coloneqq c(x_i,y_j)$ とおくと，
Kantorovich 問題は $mn$ 個の変数 $\pi_{ij}$ に関する LP になる：
\begin{alignat*}{2}
  \min\quad & \textstyle\sum_{i,j} c_{ij}\,\pi_{ij} \\
  \text{s.t.}\quad
    & \textstyle\sum_{j=1}^{n}\pi_{ij} = \mu_i &\quad& (i=1,\ldots,m), \\
    & \textstyle\sum_{i=1}^{m}\pi_{ij} = \nu_j &\quad& (j=1,\ldots,n), \\
    & \pi_{ij}\ge 0 &\quad& (\forall\,i,j).
\end{alignat*}
$m+n$ 本の等式制約のうち1本は冗長である
（$\sum_i\mu_i=\sum_j\nu_j=1$ から任意の1本が残りから従う）．
したがって独立な制約は $m+n-1$ 本であり，
基底実行可能解（定義~\ref{def:bfs}）は
高々 $m+n-1$ 個の正成分を持つ．
この形の LP は\textbf{輸送問題}（transportation problem）と呼ばれる．
\end{remark}

\begin{example}{設定~\ref{set:basic}の LP 表示}{lp_star}
設定~\ref{set:basic}（$m=n=2$）では4つの変数 $\pi_{00},\pi_{01},\pi_{10},\pi_{11}$ に対し
\begin{align*}
  \min\quad & \pi_{01} + \pi_{10} \\
  \text{s.t.}\quad
    & \pi_{00} + \pi_{01} = \tfrac{3}{4}, \quad
    \pi_{10} + \pi_{11} = \tfrac{1}{4}, \\
    & \pi_{00} + \pi_{10} = \tfrac{2}{4}, \quad
    \pi_{01} + \pi_{11} = \tfrac{2}{4}, \\
    & \pi_{ij}\ge 0
\end{align*}
が得られる（$c(x,y)=|x-y|$ から $c_{00}=c_{11}=0$，$c_{01}=c_{10}=1$）．
独立な等式制約は $m+n-1=3$ 本であり，自由度は $4-3=1$．
例~\ref{ex:kantorovich_example} と同様に $\pi_{00}$ を自由変数にとると
\[
  \pi_{01}=\tfrac{3}{4}-\pi_{00},\quad
  \pi_{10}=\tfrac{2}{4}-\pi_{00},\quad
  \pi_{11}=\pi_{00}-\tfrac{1}{4},\quad
  \text{コスト}=\tfrac{5}{4}-2\pi_{00}
\]
であり，非負制約から $\pi_{00}\in[\tfrac{1}{4},\,\tfrac{2}{4}]$ となる．
実行可能領域は $\mathbb{R}^4$ 中の1次元線分であり，
2つの端点が基底実行可能解（BFS）に対応する．
コストは $\pi_{00}$ の狭義単調減少な一次関数だから，
最小値は右端の BFS で達成される．

\begin{definition}{基底実行可能解}{bfs}
LP の制約行列 $A\in\mathbb{R}^{m\times n}$ に対し，
$r=\operatorname{rank}(A)$ 本の列を選んで正則な部分行列 $B$ を作り，
対応する変数（\textbf{基底変数}）を $x_B=B^{-1}b\ge 0$，
残りの変数（\textbf{非基底変数}）を $x_N=0$ で定めた実行可能解を
\textbf{基底実行可能解}（basic feasible solution, BFS）と呼ぶ．
BFS は実行可能多面体の\textbf{頂点}に対応する．

輸送問題では，BFS は高々 $m+n-1$ 個の正成分を持つ
輸送計画であり，
正の配分がある $(i,j)$ の集合を\textbf{基底}と呼ぶ．
\end{definition}

\begin{theorem}{LP の基本定理}{lp_fundamental}
LP の実行可能領域が非空であり，目的関数が下に有界ならば，
最適値は少なくとも一つの BFS で達成される．
\end{theorem}

\begin{proof}[証明の概略]
実行可能解 $x$ が BFS でないとする．
このとき非基底変数のうち正であるもの $x_j>0$（$j\notin B$）が存在する．
$Ax=b$ の列ベクトルの一次従属性から，
$x$ をこの方向に移動しても実行可能性を保てるベクトル $d$ が存在し，
$c^\top d\le 0$ となるよう選べる（$c^\top d>0$ ならば $-d$ を取る）．
$x+\theta d$（$\theta\ge 0$）に沿って進むと
目的関数値 $c^\top(x+\theta d)=c^\top x+\theta\,c^\top d$ は改善（または維持）され，
非負制約 $x+\theta d\ge 0$ により $\theta$ の上限で
ある成分が $0$ になる——正の変数の数が1つ減る．
この操作を繰り返すと有限ステップで BFS に到達し，
途中の目的関数値は悪化しないため，
得られた BFS は元の $x$ 以上に良い目的関数値を持つ．
よって最適値は BFS で達成される．
\end{proof}

直感的には，線形関数は凸多面体の内部で極値を取らないため，
最適値は必ず頂点（BFS）に現れる．
定理~\ref{thm:lp_fundamental}により最適解の探索は有限個の BFS の比較に帰着する．
\textbf{単体法}はこの BFS を効率的に巡回し，
各ステップで目的関数値を改善（または維持）しながら最適解に到達するアルゴリズムである．

輸送問題の LP 双対は
\[
  \max\;\textstyle\sum_i \mu_i\,u_i + \sum_j \nu_j\,v_j
  \quad\text{s.t.}\quad u_i + v_j \le c_{ij}\;\;(\forall\,i,j)
\]
であり，これは Kantorovich 双対（定理~\ref{thm:kantorovich_duality}）の
離散版にほかならない（$u_i=\varphi(x_i)$，$v_j=\psi(y_j)$）．
以下に示す輸送単体法は，
主問題の最適輸送計画と双対問題の最適対 $(u_i,v_j)$ を\textbf{同時に}求める．
各ステップにおける双対変数の計算（Step 1）と
被約費用による最適性判定（Step 2）は，
この LP 双対構造に基づいている．

\medskip
\begin{center}
{\renewcommand{\arraystretch}{1.4}
\begin{tabular}{ll}
  \hline
  \textbf{輸送単体法} & \textbf{Kantorovich 双対理論} \\
  \hline
  双対変数 $u_i,\,v_j$ & 双対関数 $\varphi(x_i),\,\psi(y_j)$ \\
  被約費用 $\bar{c}_{ij}\ge 0$ & 双対許容条件 $\varphi+\psi\le c$ \\
  基底条件 $u_i+v_j=c_{ij}$ & 相補性条件 \\
  目的関数値 $\sum\mu_i u_i+\sum\nu_j v_j$ & 双対目的関数 $\int\varphi\,d\mu+\int\psi\,d\nu$ \\
  \hline
\end{tabular}}
\end{center}

\begin{algorithm}{alg:transport_simplex}
\textbf{輸送単体法}（Transportation Simplex Method）

\medskip
\textbf{入力：}供給量 $\mu_1,\ldots,\mu_m$，需要量 $\nu_1,\ldots,\nu_n$，
コスト行列 $(c_{ij})$．

\medskip
\textbf{Step 0}（初期 BFS）\textbf{：}
初期 BFS $\{\pi_{ij}\}$ と基底 $\mathcal{B}$（$|\mathcal{B}|=m+n-1$）を構成する
（例えば\textbf{北西角法}：セル $(1,1)$ から順に
$\pi_{ij}=\min(\text{残余供給量},\,\text{残余需要量})$ を
行・列の添字の小さい順に割り当てる）．

\medskip
\textbf{Step 1}（双対変数の計算）\textbf{：}
$u_1=0$ と正規化し，基底セル $(i,j)\in\mathcal{B}$ に対する方程式
\[
  u_i + v_j = c_{ij}
\]
を解いて $u_1,\ldots,u_m$ と $v_1,\ldots,v_n$ を求める．

\medskip
\textbf{Step 2}（最適性判定）\textbf{：}
非基底セルの\textbf{被約費用}（reduced cost）
$\bar{c}_{ij} \coloneqq c_{ij} - u_i - v_j$
を計算する．
$\bar{c}_{ij}\ge 0$ が全ての非基底セルで成り立てば，
現在の BFS は最適——\textbf{終了}．

\medskip
\textbf{Step 3}（ピボット操作）\textbf{：}
$\bar{c}_{pq}<0$ となるセル $(p,q)$ を入基底セルとして選ぶ．
$(p,q)$ と基底セルを交互に結ぶ\textbf{閉路}を見つけ，
閉路上のセルに $+\theta$ と $-\theta$ を交互に加える
（$(p,q)$ には $+\theta$）．
$\theta=\min\{\pi_{ij}: (i,j) \text{ は閉路上で } {-\theta} \text{ が加わるセル}\}$
とし，$\pi_{ij}=0$ となったセルを基底から除く．
Step 1 に戻る．
\end{algorithm}

\begin{example}{設定~\ref{set:basic}を輸送単体法で解く}{simplex_star}
設定~\ref{set:basic}に輸送単体法（Algorithm~\ref{alg:transport_simplex}）を適用する．

\medskip
\textbf{反復 0：初期 BFS．}
基底 $\mathcal{B}=\{(0,0),(0,1),(1,0)\}$ に対応する以下の BFS から出発する：
\[
  \pi^{(0)} = \begin{pmatrix} 1/4 & 2/4 \\ 1/4 & 0 \end{pmatrix},
  \qquad \text{コスト}=\pi_{01}+\pi_{10}=\tfrac{3}{4}.
\]
これは例~\ref{ex:lp_star} の BFS A（実行可能領域の左端 $\pi_{00}=\tfrac{1}{4}$）に相当する．

\medskip
\textbf{Step 1：双対変数の計算．}
$u_0=0$ として基底セルの方程式 $u_i+v_j=c_{ij}$ を解く：
\[
  u_0+v_0=0 \;\Rightarrow\; v_0=0,\qquad
  u_0+v_1=1 \;\Rightarrow\; v_1=1,\qquad
  u_1+v_0=1 \;\Rightarrow\; u_1=1.
\]

\textbf{Step 2：被約費用の計算．}
唯一の非基底セル $(1,1)$ に対し
\[
  \bar{c}_{11}=c_{11}-u_1-v_1=0-1-1=-2<0.
\]
$\bar{c}_{11}<0$ であるから現在の BFS は最適でない．
セル $(1,1)$ を基底に加えることでコストが改善できる．

\medskip
\textbf{Step 3：ピボット操作．}
入基底セル $(1,1)$ を含む閉路を構成する：

  = \begin{pmatrix} 2/4 & 1/4 \\ 0 & 1/4 \end{pmatrix}.
\]
$\pi_{10}=0$ となったので $(1,0)$ を基底から除き，$(1,1)$ を基底に加える．
新しい基底は $\mathcal{B}'=\{(0,0),(0,1),(1,1)\}$，
コストは $\tfrac{1}{4}+0=\tfrac{1}{4}$ に改善された
（コストの減少分は $|\bar{c}_{11}|\times\theta=2\times\tfrac{1}{4}=\tfrac{1}{2}$）．

\[
  u_0+v_0=0,\quad u_0+v_1=1,\quad u_1+v_1=0
  \quad\Rightarrow\quad
  u_0=0,\; v_0=0,\; v_1=1,\; u_1=-1.
\]

$\bar{c}_{10}=c_{10}-u_1-v_0=1-(-1)-0=2\ge 0$．
全ての被約費用が非負であるから $\pi^{(1)}$ は\textbf{最適}である．
最適輸送コストは $\tfrac{1}{4}$ であり，
例~\ref{ex:kantorovich_example} の結果と一致する．

\begin{example}{3点$\times$2点の輸送問題}{simplex_3x2}
より大きな問題で輸送単体法の動作を確認する．
$X=\{0,1,2\}$，$Y=\{0,1\}$ とし，
\[
  \mu = \tfrac{1}{3}\delta_0 + \tfrac{1}{3}\delta_1 + \tfrac{1}{3}\delta_2,\quad
  \nu = \tfrac{1}{2}\delta_0 + \tfrac{1}{2}\delta_1,\quad
  (c_{ij}) = \begin{pmatrix} 5 & 1 \\ 1 & 4 \\ 3 & 2 \end{pmatrix}
\]
とする．基底セルは $m+n-1=4$ 個である．

\medskip
\textbf{Step 0：北西角法．}
左上 $(0,0)$ から順に $\pi_{ij}=\min(\text{残余供給},\text{残余需要})$ を割り当てると
\[
  \pi^{(0)} = \begin{pmatrix} 1/3 & 0 \\ 1/6 & 1/6 \\ 0 & 1/3 \end{pmatrix}\!,
  \quad \mathcal{B}=\{(0,0),(1,0),(1,1),(2,1)\},
  \quad \text{コスト}=\tfrac{19}{6}.
\]

\textbf{反復 1．}
双対変数：$(u_0,u_1,u_2)=(0,-4,-6)$，$(v_0,v_1)=(5,8)$．
被約費用：$\bar{c}_{01}=1-0-8=-7<0$，$\bar{c}_{20}=3+6-5=4\ge 0$．
$(0,1)$ が入基底セル，
閉路 $(0,1)\to(1,1)\to(1,0)\to(0,0)$ で
$\theta=\min(\pi_{11},\pi_{00})=\tfrac{1}{6}$：
\[
  \pi^{(1)} = \begin{pmatrix} 1/6 & 1/6 \\ 1/3 & 0 \\ 0 & 1/3 \end{pmatrix}\!,
  \quad \text{コスト}=2.
\]

\textbf{反復 2．}
双対変数：$(u_0,u_1,u_2)=(0,-4,1)$，$(v_0,v_1)=(5,1)$．
被約費用：$\bar{c}_{11}=4+4-1=7\ge 0$，$\bar{c}_{20}=3-1-5=-3<0$．
$(2,0)$ が入基底セル，
閉路 $(2,0)\to(2,1)\to(0,1)\to(0,0)$ で
$\theta=\min(\pi_{21},\pi_{00})=\tfrac{1}{6}$：
\[
  \pi^{(2)} = \begin{pmatrix} 0 & 1/3 \\ 1/3 & 0 \\ 1/6 & 1/6 \end{pmatrix}\!,
  \quad \text{コスト}=\tfrac{3}{2}.
\]

\textbf{最適性確認．}
双対変数：$(u_0,u_1,u_2)=(0,-1,1)$，$(v_0,v_1)=(2,1)$．
被約費用：$\bar{c}_{00}=5-0-2=3\ge 0$，$\bar{c}_{11}=4+1-1=4\ge 0$．
全て非負であり $\pi^{(2)}$ は最適，コストは $\tfrac{3}{2}$．

設定~\ref{set:basic}の 2$\times$2 の場合（例~\ref{ex:simplex_star}）では
1回のピボットで最適に到達したが，
この 3$\times$2 の例では異なる構造の閉路を経て2回のピボットが必要であり，
問題サイズが大きくなるにつれ系統的な単体法の重要性が増すことがわかる．
\end{example}

\begin{remark}{双対変数の一致の確認}{lp_kantorovich_dual}
例~\ref{ex:simplex_star} で得た最適双対変数
$(u_0,u_1,v_0,v_1)=(0,-1,0,1)$ は，
例~\ref{ex:duality_verification} の最適双対対
$\varphi^*(0)=0$，$\varphi^*(1)=-1$，$\psi^*(0)=0$，$\psi^*(1)=1$
に完全に一致する．
これは輸送単体法が Kantorovich 双対の最適対を
主問題の解法の副産物として自動的に算出することの具体的な確認である．
\end{remark}

\begin{remark}{Wasserstein 距離への展望}{wasserstein}
コスト $c(x,y)=d(x,y)^p$（$d$ は距離，$p\ge 1$）に対する
Kantorovich 問題の最適値の $p$ 乗根
\[
  W_p(\mu,\nu)
  \coloneqq \left(\inf_{\pi\in\Pi(\mu,\nu)}
  \int_{X\times Y} d(x,y)^p\,d\pi(x,y)\right)^{1/p}
\]
は，$X$ 上の確率測度全体の空間 $\mathcal{P}(X)$ 上の\textbf{距離}を定める．
これを\textbf{Wasserstein 距離}（$p$-Wasserstein distance）と呼ぶ．
$W_p$ が距離の公理を満たすこと——特に三角不等式——は
最適輸送計画の「貼り合わせ」（gluing lemma）によって証明される．

本稿で扱った設定~\ref{set:basic}では $c=|\cdot|$（$d=|\cdot|$，$p=1$）であるから，
最適輸送コスト $\tfrac{1}{4}$ はそのまま
$W_1(\mu,\nu)=\tfrac{1}{4}$ を与える．
Wasserstein 距離は確率分布の幾何学的な「近さ」を測る道具として，
機械学習（生成モデル），統計学（分布の検定），偏微分方程式（勾配流）
など広範な分野で中心的な役割を果たしている．
最適輸送理論のさらなる展開については
Villani~\cite{Villani2009} や Santambrogio~\cite{Santambrogio2015} を参照されたい．
\end{remark}
