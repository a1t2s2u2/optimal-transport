\begin{definition}{СйЇуЏИуЕ║жќЊ}{topological_space}
жЏєтљѕ $X$ сЂ«жЃетѕєжЏєтљѕТЌЈ $\mathcal{O}\subset 2^X$ сЂї
$X$ СИісЂ«\textbf{СйЇуЏИ}№╝ѕtopology№╝ЅсЂДсЂѓсѓІсЂесЂ»№╝ї
ТгАсЂ«СИЅТЮАС╗ХсѓњТ║ђсЂЪсЂЎсЂЊсЂесѓњсЂёсЂє№╝џ
\begin{enumerate}
  \item $\emptyset\in\mathcal{O}$ сЂІсЂц $X\in\mathcal{O}$№╝ї
  \item $O_1,O_2\in\mathcal{O}$ сЂфсѓЅсЂ░ $O_1\cap O_2\in\mathcal{O}$№╝ѕТюЅжЎљС║цтЈЅсЂДжќЅсЂўсѓІ№╝Ѕ№╝ї
  \item $\{O_\lambda\}_{\lambda\in\Lambda}\subset\mathcal{O}$ сЂфсѓЅсЂ░
        $\bigcup_{\lambda\in\Lambda}O_\lambda\in\mathcal{O}$№╝ѕС╗╗ТёЈтљѕСйхсЂДжќЅсЂўсѓІ№╝Ѕ№╝ј
\end{enumerate}
ухё $(X,\mathcal{O})$ сѓњ\textbf{СйЇуЏИуЕ║жќЊ}сЂетЉ╝сЂ│№╝ї
$\mathcal{O}$ сЂ«тЁЃсѓњ\textbf{жќІжЏєтљѕ}сЂесЂёсЂє№╝ј
\end{definition}

\begin{definition}{$\sigma$тіаТ│ЋТЌЈ}{sigma_algebra}
жЏєтљѕ $X$ сЂ«жЃетѕєжЏєтљѕТЌЈ $\mathcal{A}\subset 2^X$ сЂї
\textbf{$\sigma$тіаТ│ЋТЌЈ}№╝ѕ$\sigma$-algebra№╝ЅсЂДсЂѓсѓІсЂесЂ»№╝ї
ТгАсЂ«СИЅТЮАС╗ХсѓњТ║ђсЂЪсЂЎсЂЊсЂесѓњсЂёсЂє№╝џ
\begin{enumerate}
  \item $X\in\mathcal{A}$№╝ї
  \item $A\in\mathcal{A}$ сЂфсѓЅсЂ░ $A^c\in\mathcal{A}$№╝ї
  \item $A_1,A_2,\ldots\in\mathcal{A}$ сЂфсѓЅсЂ░
        $\bigcup_{n=1}^{\infty}A_n\in\mathcal{A}$№╝ј
\end{enumerate}
\end{definition}

\begin{definition}{тЈ»ТИгуЕ║жќЊ}{measurable_space}
жЏєтљѕ $X$ сЂесЂЮсЂ«СИісЂ« $\sigma$тіаТ│ЋТЌЈ $\mathcal{A}$ сЂ«ухё
$(X,\mathcal{A})$ сѓњ\textbf{тЈ»ТИгуЕ║жќЊ}сЂетЉ╝сЂХ№╝ј
$\mathcal{A}$ сЂ«тЁЃсѓњ\textbf{тЈ»ТИгжЏєтљѕ}сЂесЂёсЂє№╝ј
уЅ╣сЂФ№╝їСйЇуЏИуЕ║жќЊ $(X,\mathcal{O})$ сЂФт»ЙсЂЌ№╝ї
жќІжЏєтљѕтЁеСйЊ $\mathcal{O}$ сѓњжЃетѕєжЏєтљѕсЂесЂЌсЂдтљФсѓђ $\sigma$тіаТ│ЋТЌЈсЂ«сЂєсЂА№╝ї
Тюђт░ЈсЂ«сѓѓсЂ«сѓњ\textbf{Borel $\sigma$тіаТ│ЋТЌЈ}сЂетЉ╝сЂ│ $\mathcal{B}(X)$ сЂеУеўсЂЎ№╝ј
тЁиСйЊуџёсЂФсЂ»№╝ї$\mathcal{O}\subset\mathcal{F}$ сѓњТ║ђсЂЪсЂЎ
$\sigma$тіаТ│ЋТЌЈ $\mathcal{F}$ сѓњсЂЎсЂ╣сЂджЏєсѓЂсЂЪсЂесЂЇ№╝ї
$\mathcal{B}(X)=\bigcap\mathcal{F}$ сЂесЂЌсЂдтЙЌсѓЅсѓїсѓІ№╝ј
$\mathcal{B}(X)$ сЂ«тЁЃсѓњ\textbf{Borel жЏєтљѕ}сЂесЂёсЂє№╝ј
\end{definition}

\begin{example}{$\mathbb{R}$ СИісЂ«Borel $\sigma$тіаТ│ЋТЌЈ}{borel_real}
$X=\mathbb{R}$ сЂ«жќІжЏєтљѕсЂ»жќІтї║жќЊсѓёсЂЮсЂ«тљѕСйх№╝ѕСЙІ№╝џ$(0,1)$№╝ї$(-\infty,3)\cup(5,7)$№╝ЅсЂДсЂѓсѓІ№╝ј

\end{example}

\begin{definition}{ТИгт║д}{measure}
тЈ»ТИгуЕ║жќЊ $(X,\mathcal{A})$ СИісЂ«\textbf{ТИгт║д}сЂесЂ»№╝ї
жќбТЋ░ $\mu\colon\mathcal{A}\to[0,\infty]$ сЂДсЂѓсЂБсЂдТгАсѓњТ║ђсЂЪсЂЎсѓѓсЂ«сѓњсЂёсЂє№╝џ
\begin{enumerate}
  \item $\mu(\emptyset)=0$№╝ї
  \item С║њсЂёсЂФу┤асЂфтЈ»ТИгжЏєтљѕсЂ«тѕЌ $A_1,A_2,\ldots\in\mathcal{A}$
        №╝ѕ$A_i\cap A_j=\emptyset$, $i\neq j$№╝ЅсЂФт»ЙсЂЌ
        \[
          \mu\!\left(\bigcup_{n=1}^{\infty}A_n\right)
          =\sum_{n=1}^{\infty}\mu(A_n)
        \]
        сЂїТѕљсѓіуФІсЂц№╝ѕ$\sigma$тіаТ│ЋТђД№╝Ѕ№╝ј
\end{enumerate}
ухё $(X,\mathcal{A},\mu)$ сѓњ\textbf{ТИгт║дуЕ║жќЊ}сЂесЂёсЂє№╝ј
уЅ╣сЂФ $\mu(X)=1$ сѓњТ║ђсЂЪсЂЎТИгт║дсѓњ\textbf{уб║ујЄТИгт║д}сЂетЉ╝сЂХ№╝ј
$\sigma$тіаТ│ЋТЌЈсЂесЂЌсЂд Borel $\sigma$тіаТ│ЋТЌЈ $\mathcal{B}(X)$ сѓњжЂИсѓЊсЂа
тЈ»ТИгуЕ║жќЊ $(X,\mathcal{B}(X))$ СИісЂ«уб║ујЄТИгт║дсѓњ\textbf{Borelуб║ујЄТИгт║д}сЂесЂёсЂє№╝ј
\end{definition}

\begin{definition}{тЈ»ТИгтєЎтЃЈ}{measurable_map}
$(X,\mathcal{F})$№╝ї$(Y,\mathcal{G})$ сѓњтЈ»ТИгуЕ║жќЊсЂесЂЎсѓІ№╝ј
тєЎтЃЈ $T\colon X\to Y$ сЂї\textbf{тЈ»ТИгтєЎтЃЈ}сЂДсЂѓсѓІсЂесЂ»№╝ї
С╗╗ТёЈсЂ« $B\in\mathcal{G}$ сЂФт»ЙсЂЌ $T^{-1}(B)\in\mathcal{F}$ сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
\end{definition}

ТИгт║дУФќсЂДсЂ»№╝їжЏєтљѕсЂ«сђїтцДсЂЇсЂЋсђЇсЂ» $\sigma$тіаТ│ЋТЌЈсЂФт▒ъсЂЎсѓІжЏєтљѕсЂФт»ЙсЂЌсЂдсЂ«сЂ┐т«џсЂЙсѓІ№╝ј
тєЎтЃЈ $T$ сѓњжђџсЂўсЂд $Y$ тЂ┤сЂ«жЏєтљѕ $B$ сѓњ $X$ тЂ┤сЂДТЅ▒сЂєсЂФсЂ»№╝ї
жђєтЃЈ $T^{-1}(B)$ сЂї $X$ сЂ« $\sigma$тіаТ│ЋТЌЈ $\mathcal{A}$ сЂФт▒ъсЂЎсѓІт┐ЁУдЂсЂїсЂѓсѓІ№╝ј

\begin{definition}{BorelтєЎтЃЈ}{borel_map}
$X$№╝ї$Y$ сѓњСйЇуЏИуЕ║жќЊсЂесЂЎсѓІ№╝ј
тєЎтЃЈ $T\colon X\to Y$ сЂї\textbf{BorelтєЎтЃЈ}№╝ѕBorelтЈ»ТИгтєЎтЃЈ№╝ЅсЂДсЂѓсѓІсЂесЂ»№╝ї
$T$ сЂїтЈ»ТИгуЕ║жќЊ $(X,\mathcal{B}(X))$ сЂІсѓЅ $(Y,\mathcal{B}(Y))$ сЂИсЂ«тЈ»ТИгтєЎтЃЈсЂДсЂѓсѓІсЂЊсЂе№╝ї
сЂЎсЂфсѓЈсЂАС╗╗ТёЈсЂ« $B\in\mathcal{B}(Y)$ сЂФт»ЙсЂЌ $T^{-1}(B)\in\mathcal{B}(X)$ сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
\end{definition}

\begin{definition}{Ті╝сЂЌтЄ║сЂЌТИгт║д}{pushforward}
$(X,\mathcal{F})$№╝ї$(Y,\mathcal{G})$ сѓњтЈ»ТИгуЕ║жќЊ№╝ї
$\mu$ сѓњ $(X,\mathcal{F})$ СИісЂ«ТИгт║д№╝ї$T\colon X\to Y$ сѓњтЈ»ТИгтєЎтЃЈсЂесЂЎсѓІ№╝ј
$T$ сЂФсѓѕсѓІ $\mu$ сЂ«\textbf{Ті╝сЂЌтЄ║сЂЌТИгт║д} $T_\#\mu$ сѓњ
\[
  T_\#\mu(B) \coloneqq \mu(T^{-1}(B)), \qquad B\in\mathcal{G}
\]
сЂДт«џуЙЕсЂЎсѓІ№╝ј
\end{definition}

\begin{definition}{Monge тЋЈжАї}{monge}
$X$№╝ї$Y$ сѓњСйЇуЏИуЕ║жќЊсЂесЂЌ№╝ї
$\mu$№╝ї$\nu$ сѓњсЂЮсѓїсЂъсѓї $X$№╝ї$Y$ СИісЂ«Borelуб║ујЄТИгт║д№╝ї
$c\colon X\times Y\to[0,\infty]$ сѓњBorelтєЎтЃЈсЂесЂЎсѓІ№╝ј
\textbf{Monge тЋЈжАї}сЂ»№╝ї$T_\#\mu=\nu$ сѓњТ║ђсЂЪсЂЎBorelтєЎтЃЈ

\[

сѓњжЂћТѕљсЂЎсѓІсѓѓсЂ«сѓњТ▒ѓсѓЂсѓІтЋЈжАїсЂДсЂѓсѓІ№╝ј
\end{definition}

\begin{definition}{DiracТИгт║д}{dirac}
$X$ сѓњСйЇуЏИуЕ║жќЊ№╝ї$x_0\in X$ сЂесЂЎсѓІ№╝ј
С╗╗ТёЈсЂ«BorelжЏєтљѕ $B\in\mathcal{B}(X)$ сЂФт»ЙсЂЌ
\[
  \delta_{x_0}(B) \coloneqq
  \begin{cases}
    1 & (x_0\in B), \\
    0 & (x_0\notin B)
  \end{cases}
\]
сЂДт«џсЂЙсѓІBorelуб║ујЄТИгт║д $\delta_{x_0}$ сѓњ№╝ї
$x_0$ сЂФжЏєСИГсЂЎсѓІ\textbf{DiracТИгт║д}сЂесЂёсЂє№╝ј
\end{definition}

\begin{remark}{DiracТИгт║дсЂ»Borelуб║ујЄТИгт║дсЂДсЂѓсѓІ}{dirac_is_borel}
DiracТИгт║д $\delta_{x_0}$ сЂ»Borelуб║ујЄТИгт║дсЂДсЂѓсѓІсЂЊсЂесѓњуб║УфЇсЂЎсѓІ№╝ј
\begin{itemize}
  \item $\delta_{x_0}(\emptyset)=0$№╝џ$x_0\notin\emptyset$ сѓѕсѓіТѕљуФІ№╝ј
  \item $\sigma$тіаТ│ЋТђД№╝џС║њсЂёсЂФу┤асЂфBorelжЏєтљѕсЂ«тѕЌ $B_1,B_2,\ldots\in\mathcal{B}(X)$ сЂФт»ЙсЂЌ№╝ї
        $x_0$ сЂ»сЂЪсЂІсЂасЂІСИђсЂцсЂ« $B_n$ сЂФсЂЌсЂІт▒ъсЂЋсЂфсЂёсЂ«сЂД№╝ї
        $\delta_{x_0}(\bigcup_n B_n)$ сЂе $\sum_n\delta_{x_0}(B_n)$ сЂ»
        сЂесѓѓсЂФ $x_0$ сЂїсЂёсЂџсѓїсЂІсЂ« $B_n$ сЂФт▒ъсЂЎсѓїсЂ░ $1$№╝їсЂЮсЂєсЂДсЂфсЂЉсѓїсЂ░ $0$ сЂДСИђУЄ┤сЂЎсѓІ№╝ј
  \item $\delta_{x_0}(X)=1$№╝џ$x_0\in X$ сѓѕсѓіТѕљуФІ№╝ѕуб║ујЄТИгт║дсЂ«ТЮАС╗Х№╝Ѕ№╝ј
\end{itemize}
сЂЌсЂЪсЂїсЂБсЂд $\delta_{x_0}$ сЂ»Borelуб║ујЄТИгт║дсЂДсЂѓсѓІ№╝ј
\end{remark}

\begin{proposition}{DiracТИгт║дсЂФжќбсЂЎсѓІуЕЇтѕє}{integral_dirac}
$f\colon X\to\mathbb{R}$ сѓњBorelтєЎтЃЈсЂесЂЎсѓІ№╝ј
DiracТИгт║д $\delta_{x_0}$ сЂФжќбсЂЎсѓІуЕЇтѕєсЂ»
\[
  \int_X f(x)\,d\delta_{x_0}(x) = f(x_0)
\]
сЂДСИјсЂѕсѓЅсѓїсѓІ№╝ј
\end{proposition}

\begin{definition}{У╝ИжђЂУеѕућ╗}{transport_plan}
$X$№╝ї$Y$ сѓњСйЇуЏИуЕ║жќЊсЂесЂЌ№╝ї
$\mu$№╝ї$\nu$ сѓњсЂЮсѓїсЂъсѓї $X$№╝ї$Y$ СИісЂ«Borelуб║ујЄТИгт║дсЂесЂЎсѓІ№╝ј
$X\times Y$ СИісЂ«Borelуб║ујЄТИгт║д $\pi$ сЂї
$\mu$ сЂе $\nu$ сЂ«\textbf{У╝ИжђЂУеѕућ╗}сЂДсЂѓсѓІсЂесЂ»№╝ї
$\pi$ сЂ«тЉеУЙ║тѕєтИЃсЂїсЂЮсѓїсЂъсѓї $\mu$ сЂе $\nu$ сЂФСИђУЄ┤сЂЎсѓІсЂЊсЂе№╝ї
сЂЎсЂфсѓЈсЂАС╗╗ТёЈсЂ«BorelжЏєтљѕ $A\subset X$№╝ї$B\subset Y$ сЂФт»ЙсЂЌ
\[
  \pi(A\times Y) = \mu(A), \qquad
  \pi(X\times B) = \nu(B)
\]
сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
$\mu$ сЂе $\nu$ сЂ«У╝ИжђЂУеѕућ╗тЁеСйЊсЂ«жЏєтљѕсѓњ $\Pi(\mu,\nu)$ сЂеУеўсЂЎ№╝ј
\end{definition}

жЏбТЋБуџёсЂфта┤тљѕ№╝ї$\pi$ сѓњУАїтѕЌ $(\pi_{xy})$ сЂДУАесЂЎсЂе
тљёТѕљтѕє $\pi_{xy}$ сЂ»сђїтЄ║уЎ║тю░ $x$ сЂФсЂѓсѓІтюЪсѓњтѕ░жЂћтю░ $y$ сЂИУ╝ИжђЂсЂЎсѓІжЄЈсђЇсѓњТёЈтЉ│сЂЎсѓІ№╝ј
\begin{itemize}
  \item \textbf{УАї $x$ сЂ«тњї}сЂї $\mu(\{x\})$ сЂФСИђУЄ┤сЂЎсѓІТЮАС╗ХсЂ»№╝ї
        сђїтЄ║уЎ║тю░ $x$ сЂІсѓЅжђЂсѓітЄ║сЂЎтюЪсЂ«уиЈжЄЈсЂїсЂАсѓЄсЂєсЂЕСЙЏухджЄЈсЂФуГЅсЂЌсЂёсђЇсЂЊсЂесѓњУдЂУФІсЂЎсѓІ№╝ј
  \item \textbf{тѕЌ $y$ сЂ«тњї}сЂї $\nu(\{y\})$ сЂФСИђУЄ┤сЂЎсѓІТЮАС╗ХсЂ»№╝ї
        сђїтѕ░жЂћтю░ $y$ сЂФтЈЌсЂЉтЈќсѓІтюЪсЂ«уиЈжЄЈсЂїсЂАсѓЄсЂєсЂЕжюђУдЂжЄЈсЂФуГЅсЂЌсЂёсђЇсЂЊсЂесѓњУдЂУФІсЂЎсѓІ№╝ј
  \item т»ЙУДњТѕљтѕє $\pi_{xx}$ сЂ»сЂЮсЂ«та┤сЂФ\textbf{уЋЎсЂЙсѓІ}жЄЈ№╝ї
        жЮът»ЙУДњТѕљтѕє $\pi_{xy}\ (x\ne y)$ сЂ» $x$ сЂІсѓЅ $y$ сЂИ\textbf{уД╗тІЋсЂЎсѓІ}жЄЈсЂФт»Йт┐юсЂЎсѓІ№╝ј
\end{itemize}

\begin{definition}{Kantorovich тЋЈжАї}{kantorovich}
$X$№╝ї$Y$ сѓњСйЇуЏИуЕ║жќЊсЂесЂЌ№╝ї
$\mu$№╝ї$\nu$ сѓњсЂЮсѓїсЂъсѓї $X$№╝ї$Y$ СИісЂ«Borelуб║ујЄТИгт║д№╝ї
$c\colon X\times Y\to[0,\infty]$ сѓњBorelтєЎтЃЈсЂесЂЎсѓІ№╝ј
\textbf{Kantorovich тЋЈжАї}сЂ»
\[
  \inf_{\pi\in\Pi(\mu,\nu)}
  \int_{X\times Y} c(x,y)\,d\pi(x,y)
\]
сѓњТ▒ѓсѓЂсѓІтЋЈжАїсЂДсЂѓсѓІ№╝ј
\end{definition}

\begin{remark}{Monge тЋЈжАїсЂесЂ«жќбС┐ѓ}{monge_kantorovich}
Monge сЂ«У╝ИжђЂтєЎтЃЈ $T\colon X\to Y$ сЂїСИјсЂѕсѓЅсѓїсЂЪсЂесЂЇ№╝ї
тєЎтЃЈ $x\mapsto(x,\,T(x))$ сЂФсѓѕсѓІ $\mu$ сЂ«Ті╝сЂЌтЄ║сЂЌТИгт║д
\[
  \pi(A\times B) = \mu(\{x\in A : T(x)\in B\})
\]
сЂ»У╝ИжђЂУеѕућ╗сЂФсЂфсѓІ№╝ј
сЂЊсѓїсЂ»сђї$x$ сѓњ $T(x)$ сЂФуб║т«џуџёсЂФжђЂсѓІсђЇсЂесЂёсЂєУеѕућ╗сѓњ№╝ї
$X\times Y$ СИісЂ«тљїТЎѓтѕєтИЃсЂесЂЌсЂдУАеуЈЙсЂЌсЂЪсѓѓсЂ«сЂДсЂѓсѓІ№╝ј
сЂЌсЂЪсЂїсЂБсЂд Kantorovich тЋЈжАїсЂ» Monge тЋЈжАїсЂ«уюЪсЂ«уиЕтњїсЂДсЂѓсѓі№╝ї
Kantorovich тЋЈжАїсЂ«ТюђжЂЕтђцсЂ» Monge тЋЈжАїсЂ«ТюђжЂЕтђцС╗ЦСИІсЂДсЂѓсѓІ№╝ј
сЂЋсѓЅсЂФ $\Pi(\mu,\nu)$ сЂ»тИИсЂФжЮъуЕ║сЂДсЂѓсѓІ№╝ј
т«ЪжџЏ№╝ї$X\times Y$ СИісЂ«ТИгт║д $\pi$ сѓњ
\[
  \pi(A\times B) \coloneqq \mu(A)\cdot\nu(B)
\]
сЂДт«џсѓЂсѓІсЂе№╝їтЉеУЙ║тѕєтИЃТЮАС╗ХсѓњТ║ђсЂЪсЂЎ№╝џ
\begin{align*}
  \pi(A\times Y) &= \mu(A)\cdot\nu(Y) = \mu(A)\cdot 1 = \mu(A), \\
  \pi(X\times B) &= \mu(X)\cdot\nu(B) = 1\cdot\nu(B) = \nu(B).
\end{align*}
сѓѕсЂБсЂд $\pi\in\Pi(\mu,\nu)$ сЂДсЂѓсѓі№╝ї$\Pi(\mu,\nu)$ сЂ»тИИсЂФжЮъуЕ║сЂДсЂѓсѓІ№╝ј
сЂЌсЂЪсЂїсЂБсЂд№╝їMonge тЋЈжАїсЂесЂ»уЋ░сЂфсѓі Kantorovich тЋЈжАїсЂ»тИИсЂФт«ЪУАїтЈ»УЃйсЂДсЂѓсѓІ№╝ј
\end{remark}

$\pi$ сЂ»4уѓ╣СИісЂ«уб║ујЄУ│фжЄЈ $\pi_{xy}\coloneqq\pi(\{x\}\times\{y\})\ge0$ сЂДт«їтЁесЂФТ▒║сЂЙсѓі№╝ї
$2\times2$ УАїтѕЌ
\[
  \pi = \begin{pmatrix} \pi_{00} & \pi_{01} \\ \pi_{10} & \pi_{11} \end{pmatrix}
\]
сЂДУАесЂЏсѓІ№╝јУ╝ИжђЂУеѕућ╗сЂ«т«џуЙЕ№╝ѕт«џуЙЕ~\ref{def:transport_plan}№╝ЅсЂ«тЉеУЙ║тѕєтИЃТЮАС╗Х
$\pi(A\times Y)=\mu(A)$ сЂісѓѕсЂ│ $\pi(X\times B)=\nu(B)$ сѓњ№╝ї
$A$ сЂе $B$ сЂесЂЌсЂдсЂЮсѓїсЂъсѓї $\{0\}$ сЂе $\{1\}$ сЂФжЂЕућесЂЎсѓІсЂе№╝ї
\begin{align}
  \pi(\{0\}\times Y) &= \pi_{00}+\pi_{01} = \mu(\{0\}) = \tfrac{3}{4} \label{eq:marg_mu0}\\
  \pi(\{1\}\times Y) &= \pi_{10}+\pi_{11} = \mu(\{1\}) = \tfrac{1}{4} \label{eq:marg_mu1}\\
  \pi(X\times\{0\}) &= \pi_{00}+\pi_{10} = \nu(\{0\}) = \tfrac{2}{4} \label{eq:marg_nu0}\\
  \pi(X\times\{1\}) &= \pi_{01}+\pi_{11} = \nu(\{1\}) = \tfrac{2}{4} \label{eq:marg_nu1}
\end{align}
сЂЎсЂфсѓЈсЂА№╝їсђїУАїсЂ«тњїсЂї $\mu$№╝їтѕЌсЂ«тњїсЂї $\nu$ сЂФСИђУЄ┤сЂЎсѓІжЮъУ▓аУАїтѕЌсђЇсЂї
$\Pi(\mu,\nu)$ сЂ«тЁЃсЂДсЂѓсѓІ№╝ј

\medskip
\textbf{ТюђжЂЕУеѕућ╗сЂ«т░јтЄ║№╝ј}
$c(x,y)=|x-y|$ сѓњтљёТа╝тГљуѓ╣сЂДУЕЋСЙАсЂЎсѓІсЂе№╝ї
У╝ИжђЂсѓ│сѓ╣сЃѕсЂ»
\[
  \int c\,d\pi
  = 0\cdot\pi_{00} + 1\cdot\pi_{01} + 1\cdot\pi_{10} + 0\cdot\pi_{11}
  = \pi_{01} + \pi_{10}
\]
сЂеТЏИсЂЉсѓІ№╝ј\eqref{eq:marg_mu0}№╝ї\eqref{eq:marg_nu0} сЂІсѓЅ

сЂесЂфсѓі№╝ї$\pi_{00}$ сѓњТюђтцДтїќсЂЎсѓІсЂЊсЂесЂДТюђт░ЈтїќсЂЋсѓїсѓІ№╝ј
$\pi_{00}\ge0$ сЂІсЂцтљётЉеУЙ║тѕєтИЃТЮАС╗ХсЂІсѓЅ
$\pi_{00}\le\min\!\bigl(\tfrac{3}{4},\,\tfrac{2}{4}\bigr)=\tfrac{2}{4}$ сЂфсЂ«сЂД№╝ї
ТюђжЂЕУеѕућ╗сЂ» $\pi_{00}=\tfrac{2}{4}$ сЂДСИјсЂѕсѓЅсѓї№╝ї
\[
  \pi^* = \begin{pmatrix} 2/4 & 1/4 \\ 0 & 1/4 \end{pmatrix},
  \qquad
  \inf_{\pi\in\Pi(\mu,\nu)}\int c\,d\pi
  = \tfrac{5}{4} - \tfrac{4}{4} = \tfrac{1}{4}.
\]

\begin{definition}{сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊ}{polish_space}
т«їтѓЎсЂІсЂцтЈ»тѕєсЂфУиЮжЏбуЕ║жќЊсѓњ\textbf{сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊ}сЂетЉ╝сЂХ№╝ј
\end{definition}

\begin{definition}{СИІтЇіжђБуХџ}{lsc}
УиЮжЏбуЕ║жќЊ $X$ СИісЂ«жќбТЋ░ $f\colon X\to(-\infty,\infty]$ сЂї
\textbf{СИІтЇіжђБуХџ}сЂДсЂѓсѓІсЂесЂ»№╝ї
С╗╗ТёЈсЂ« $x\in X$ сЂе $x$ сЂФтЈјТЮЪсЂЎсѓІуѓ╣тѕЌ $x_n\to x$ сЂФт»ЙсЂЌ
\[
  f(x) \le \liminf_{n\to\infty} f(x_n)
\]
сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
уЏ┤ТёЪуџёсЂФсЂ»№╝їжќбТЋ░тђцсЂїсђїСИІсЂФУи│сЂХсђЇсЂЊсЂесЂ»сЂѓсЂБсЂдсѓѓсђїСИІсЂФУљйсЂАсѓІсђЇсЂЊсЂесЂїсЂфсЂёсЂесЂёсЂєТЮАС╗ХсЂДсЂѓсѓІ№╝ј
\end{definition}

\begin{definition}{ТИгт║дсЂ«т╝▒тЈјТЮЪ}{weak_convergence}
сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊ $X$ СИісЂ«Borelуб║ујЄТИгт║дсЂ«тѕЌ $\{\mu_n\}$ сЂї
$\mu$ сЂФ\textbf{т╝▒тЈјТЮЪ}сЂЎсѓІсЂесЂ»№╝ї
С╗╗ТёЈсЂ«ТюЅуЋїжђБуХџжќбТЋ░ $f\colon X\to\mathbb{R}$ сЂФт»ЙсЂЌ
\[
  \lim_{n\to\infty}\int_X f\,d\mu_n = \int_X f\,d\mu
\]
сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
\end{definition}

\begin{definition}{уиіт»єТђД}{tightness}
сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊ $X$ СИісЂ«Borelуб║ујЄТИгт║дсЂ«ТЌЈ $\mathcal{M}$ сЂї\textbf{уиіт»є}№╝ѕtight№╝ЅсЂДсЂѓсѓІсЂесЂ»№╝ї
С╗╗ТёЈсЂ« $\varepsilon>0$ сЂФт»ЙсЂЌ№╝їсЂѓсѓІсѓ│сЃ│сЃЉсѓ»сЃѕжЏєтљѕ $K\subset X$ сЂїтГўтюесЂЌсЂд№╝ї
сЂЎсЂ╣сЂдсЂ« $\mu\in\mathcal{M}$ сЂФт»ЙсЂЌ $\mu(X\setminus K)<\varepsilon$ сЂїТѕљсѓіуФІсЂцсЂЊсЂесѓњсЂёсЂє№╝ј
уЏ┤ТёЪуџёсЂФсЂ»№╝їТИгт║дсЂїсђїуёАжЎљжЂасЂФжђЃсЂњсЂфсЂёсђЇсЂЊсЂесѓњТёЈтЉ│сЂЎсѓІ№╝ј
\end{definition}

\begin{theorem}{ProkhorovсЂ«т«џуљє}{prokhorov}
сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊСИісЂ«Borelуб║ујЄТИгт║дсЂ«ТЌЈ $\mathcal{M}$ сЂїуиіт»єсЂДсЂѓсѓІсЂЊсЂесЂе№╝ї
$\mathcal{M}$ сЂї\textbf{уЏИт»Йсѓ│сЃ│сЃЉсѓ»сЃѕ}
№╝ѕС╗╗ТёЈсЂ«тѕЌсЂїт╝▒тЈјТЮЪсЂЎсѓІжЃетѕєтѕЌсѓњТїЂсЂц№╝ЅсЂДсЂѓсѓІсЂЊсЂесЂ»тљїтђцсЂДсЂѓсѓІ№╝ј
\end{theorem}

\begin{proof}
№╝ѕуЏИт»Йсѓ│сЃ│сЃЉсѓ»сЃѕ $\Rightarrow$ уиіт»є№╝Ѕ
т»ЙтЂХсѓњуц║сЂЎ№╝јуиіт»єсЂДсЂфсЂёсЂеС╗«т«џсЂЎсѓІсЂе№╝ї
сЂѓсѓІ $\varepsilon>0$ сЂїтГўтюесЂЌсЂд№╝ї
С╗╗ТёЈсЂ«сѓ│сЃ│сЃЉсѓ»сЃѕжЏєтљѕ $K$ сЂФт»ЙсЂЌ $\mu_K(X\setminus K)\ge\varepsilon$
сѓњТ║ђсЂЪсЂЎ $\mu_K\in\mathcal{M}$ сЂїтЈќсѓїсѓІ№╝ј
сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊсЂДсЂ»сѓ│сЃ│сЃЉсѓ»сЃѕжЏєтљѕсЂ«тбЌтцДтѕЌ $K_1\subset K_2\subset\cdots$
сЂД $X=\bigcup_n K_n$ сЂесЂфсѓІсѓѓсЂ«сЂїтГўтюесЂЎсѓІсЂ«сЂД№╝ї
$\mu_{K_n}(X\setminus K_n)\ge\varepsilon$ сѓњТ║ђсЂЪсЂЎтѕЌ $\{\mu_{K_n}\}$ сЂїтЙЌсѓЅсѓїсѓІ№╝ј
сЂЊсЂ«тѕЌсЂ«С╗╗ТёЈсЂ«жЃетѕєтѕЌсЂ«т╝▒тЈјТЮЪтЁѕ $\mu$ сЂФт»ЙсЂЌсЂдсѓѓ
$\mu(X\setminus K_n)\ge\varepsilon$ сЂїС╗╗ТёЈсЂ« $n$ сЂДТѕљсѓіуФІсЂцсЂЊсЂесЂФсЂфсѓі№╝ї
$\mu(X)=\lim_{n\to\infty}\mu(K_n)\le 1-\varepsilon<1$ сЂесЂфсЂБсЂд
уб║ујЄТИгт║дсЂДсЂѓсѓІсЂЊсЂесЂФуЪЏуЏЙсЂЎсѓІ№╝ј
сѓѕсЂБсЂдуЏИт»Йсѓ│сЃ│сЃЉсѓ»сЃѕсЂфсѓЅсЂ░уиіт»єсЂДсЂѓсѓІ№╝ј

№╝ѕуиіт»є $\Rightarrow$ уЏИт»Йсѓ│сЃ│сЃЉсѓ»сЃѕ№╝Ѕ
$\mathcal{M}$ сЂІсѓЅС╗╗ТёЈсЂ«тѕЌ $\{\mu_n\}$ сѓњтЈќсѓІ№╝ј
сЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊсЂ»тЈ»тѕєсЂДсЂѓсѓІсЂІсѓЅ№╝ї
уеат»єсЂфтЈ»у«ЌжЏєтљѕ $\{q_1,q_2,\ldots\}$ сЂесЂЮсѓїсѓЅсѓњСИГт┐ЃсЂесЂЎсѓІ
ТюЅуљєТЋ░тЇітЙёсЂ«жќІуљЃтЁеСйЊсЂесЂёсЂєтЈ»у«ЌТЌЈсЂїжќІжЏєтљѕсЂ«тЪ║т║ЋсѓњсЂфсЂЎ№╝ј
$\mu_n$ сЂ«сЂЊсЂ«тЈ»у«ЌТЌЈСИісЂДсЂ«тђцсЂФт»ЙУДњуиџУФќТ│ЋсѓњжЂЕућесЂЎсѓІсЂе№╝ї
сЂЎсЂ╣сЂдсЂ«жќІуљЃСИісЂДТЦхжЎљсЂїт«џсЂЙсѓІжЃетѕєтѕЌ $\{\mu_{n_k}\}$ сЂїтЈќсѓїсѓІ№╝ј
уиіт»єТђДсЂФсѓѕсѓіТЦхжЎљсЂ»Borelуб║ујЄТИгт║д $\mu$ сѓњт«џсѓЂ№╝ї
$\mu_{n_k}$ сЂ» $\mu$ сЂФт╝▒тЈјТЮЪсЂЎсѓІ№╝ј
\end{proof}

\begin{theorem}{KantorovichтЋЈжАїсЂ«ТюђжЂЕУДБсЂ«тГўтюе}{kantorovich_existence}
$X,Y$ сѓњсЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊсЂесЂЌ№╝ї
$c\colon X\times Y\to[0,\infty]$ сЂїСИІтЇіжђБуХџсЂДсЂѓсѓІсЂесЂЎсѓІ№╝ј
сЂЊсЂ«сЂесЂЇ Kantorovich тЋЈжАїсЂ«СИІжЎљсЂ»Тюђт░ЈтђцсЂесЂЌсЂджЂћТѕљсЂЋсѓїсѓІ№╝ј
сЂЎсЂфсѓЈсЂА№╝їсЂѓсѓІ $\pi^*\in\Pi(\mu,\nu)$ сЂїтГўтюесЂЌсЂд
\[
  \int_{X\times Y} c(x,y)\,d\pi^*(x,y)
  = \inf_{\pi\in\Pi(\mu,\nu)}
    \int_{X\times Y} c(x,y)\,d\pi(x,y)
\]
сЂїТѕљсѓіуФІсЂц№╝ј
\end{theorem}

\begin{proof}
$\Pi(\mu,\nu)$ сЂ»тЉеУЙ║тѕєтИЃТЮАС╗ХсЂФсѓѕсѓіуиіт»є№╝ѕtight№╝ЅсЂДсЂѓсѓі№╝ї
Prokhorov сЂ«т«џуљєсЂІсѓЅТИгт║дсЂ«т╝▒тЈјТЮЪсЂФжќбсЂЌсЂдуЏИт»Йсѓ│сЃ│сЃЉсѓ»сЃѕсЂДсЂѓсѓІ№╝ј
тЉеУЙ║тѕєтИЃТЮАС╗ХсЂ»т╝▒тЈјТЮЪсЂФжќбсЂЌсЂджќЅсЂўсЂдсЂёсѓІсЂ«сЂД№╝ї$\Pi(\mu,\nu)$ сЂ»сѓ│сЃ│сЃЉсѓ»сЃѕсЂДсЂѓсѓІ№╝ј
$c$ сЂїСИІтЇіжђБуХџсЂДсЂѓсѓІсЂЊсЂесЂІсѓЅ
$\pi\mapsto\int c\,d\pi$ сЂ»т╝▒тЈјТЮЪсЂФжќбсЂЌсЂдСИІтЇіжђБуХџсЂДсЂѓсѓі№╝ї
сѓ│сЃ│сЃЉсѓ»сЃѕжЏєтљѕСИісЂ«СИІтЇіжђБуХџжќбТЋ░сЂ»Тюђт░ЈтђцсѓњжЂћТѕљсЂЎсѓІ№╝ј
\end{proof}

\begin{theorem}{Kantorovich тЈїт»Йт«џуљє}{kantorovich_duality}
$X,Y$ сѓњсЃЮсЃ╝сЃЕсЃ│сЃЅуЕ║жќЊсЂесЂЌ№╝ї
$c\colon X\times Y\to[0,\infty]$ сЂїСИІтЇіжђБуХџсЂесЂЎсѓІ№╝јсЂЊсЂ«сЂесЂЇ
\[
  \inf_{\pi\in\Pi(\mu,\nu)}
  \int_{X\times Y} c(x,y)\,d\pi(x,y)
  = \sup_{(\varphi,\psi)\in\Phi_c}
  \left(
    \int_{X}\varphi(x)\,d\mu(x)
    + \int_{Y}\psi(y)\,d\nu(y)
  \right)
\]
сЂїТѕљсѓіуФІсЂц№╝јсЂЊсЂЊсЂД
$\Phi_c = \{(\varphi,\psi)\in L^1(\mu)\times L^1(\nu)
: \varphi(x)+\psi(y)\le c(x,y)\}$
сЂ»\textbf{тЈїт»ЙУе▒т«╣т»Й}сЂ«жЏєтљѕсЂДсЂѓсѓІ№╝ј
\end{theorem}

тЈїт»ЙтЋЈжАїсЂФсЂ»Тўјт┐ФсЂфухїТИѕуџёУДБжЄѕсЂїсЂѓсѓІ№╝ј
$\varphi(x)$ сѓњтЄ║уЎ║тю░ $x$ сЂДсЂ«уЕЇсЂ┐УЙ╝сЂ┐СЙАТа╝№╝ї
$\psi(y)$ сѓњтѕ░уЮђтю░ $y$ сЂДсЂ«УЇижЎЇсѓЇсЂЌСЙАТа╝сЂеУђЃсЂѕсѓІсЂе№╝ї
тѕХу┤ё $\varphi(x)+\psi(y)\le c(x,y)$ сЂ»
УБЂт«џТЮАС╗Х№╝ѕСЙАТа╝ти«сЂїУ╝ИжђЂсѓ│сѓ╣сЃѕсѓњУХЁсЂѕсЂфсЂё№╝ЅсѓњТёЈтЉ│сЂЎсѓІ№╝ј
тЈїт»ЙтЋЈжАїсЂ»№╝їсЂЊсЂ«УБЂт«џТЮАС╗ХсЂ«СИІсЂДС╗▓С╗ІТЦГУђЁсЂ«уиЈтЈјтЁЦсѓњТюђтцДтїќсЂЎсѓІтЋЈжАїсЂДсЂѓсѓі№╝ї
т╝итЈїт»ЙТђДсЂ»ТюђжЂЕУ╝ИжђЂсѓ│сѓ╣сЃѕсЂеТюђжЂЕСЙАТа╝УеГт«џсЂФсѓѕсѓІтЈјтЁЦсЂїСИђУЄ┤сЂЎсѓІсЂЊсЂесѓњУ┐░сЂ╣сЂдсЂёсѓІ№╝ј

\begin{remark}{Monge тЋЈжАїсЂИсЂ«тЏътИ░}{monge_from_kantorovich}
Kantorovich тЋЈжАїсЂ«ТюђжЂЕУеѕућ╗ $\pi^*$ сЂї
$\pi^*=({\rm id},T)_\#\mu$ сЂ«тйбсЂФТЏИсЂЉсѓІ№╝ѕсЂЎсЂфсѓЈсЂАсѓ░сЃЕсЃЋСИісЂФжЏєСИГсЂЎсѓІ№╝Ѕта┤тљѕ№╝ї
$T$ сЂ» Monge тЋЈжАїсЂ«ТюђжЂЕтєЎтЃЈсЂДсѓѓсЂѓсѓІ№╝ј
СЙІсЂѕсЂ░ $X=Y=\mathbb{R}^d$ сЂД $c(x,y)=\|x-y\|^2$ сЂІсЂц $\mu$ сЂїухХт»ЙжђБуХџсЂ«сЂесЂЇ№╝ї
Brenier сЂ«т«џуљєсЂФсѓѕсѓіТюђжЂЕУ╝ИжђЂтєЎтЃЈ $T=\nabla\phi$
№╝ѕтЄИжќбТЋ░сЂ«тІЙжЁЇ№╝ЅсЂ«тГўтюесЂеСИђТёЈТђДсЂїС┐ЮУе╝сЂЋсѓїсѓІ№╝ј
\end{remark}
